 22: /**
 23:  * Utility class that generates unique timestamps. The resulting timestamps
 24:  * can not be treated as dates directly since they are constructed from the
 25:  * system time by appending a salt. To retrieve the system time from the
 26:  * generated timestamps use {@link #systemTime(long)}.
 27:  * <p/>
 28:  * The binary format of UniqueTimestampGenerator is part of its API and is
 29:  * guaranteed to be stable.
 30:  * <p/>
 31:  * All timestamps generated with {@link #next()} are guaranteed to come out
 32:  * in a sorted order. That is, they can be sorted as the standard Java
 33:  * {@code long} type.
 34:  */
 35: public class UniqueTimestampGenerator {
 36: 
 37:     /**
 38:      * Number of bits out of 64 used to store the system time.
 39:      */
 40:     public static final int TIME_BITS = 44;
 41: 
 42:     /**
 43:      * Number of bits out of 64 used to store the salt.
 44:      */
 45:     public static final int SALT_BITS = 64 - TIME_BITS;
 46: 
 47:     /**
 48:      * Maximum number of unique timestamps per millisecond.
 49:      */
 50:     // TODO: Shifting TIME_BITS is wrong. Fix this code
 51:     public static final long MAX_SALT = ~0 >>> TIME_BITS; // ~0 is all ones
 52: 
 53:     /**
 54:      * The maximal system time at which this class will function correctly.
 55:      */
 56:     public static final long MAX_TIME = (~MAX_SALT) >>> (SALT_BITS + 1);
 57: 
 58:     private long salt;
 59:     private long last;
 60:     private DateFormat dateFormat;
 61: 
 62:     /**
 63:      * Create a new UniqueTimestampGenerator.
 64:      * @throws AssertionError if the system time is past {@link #MAX_TIME}
 65:      */
 66:     public UniqueTimestampGenerator() {
 67:         salt = 0;
 68:         last = -1;
 69: 
 70:         if (System.currentTimeMillis () >= MAX_TIME) {
 71:             throw new AssertionError("System time past " + MAX_TIME + ". UniqueTimestampGenerator can not" +
 72:                                      " function past " + new Date(MAX_TIME));
 73:         }
 74: 
 75:         dateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSS");
 76:     }
 77: 
 78:     /**
 79:      * Return a salted timestamp that is guaranteed to be unique among all
 80:      * timestamps generated from this UniqueTimestampGenerator instance.
 81:      * <p/>
 82:      * To get the system time from the returned timestamp use the
 83:      * {@link #systemTime(long)} method.
 84:      * <p/>
 85:      * If more than {@link #MAX_SALT} timestamps has been generated within the
 86:      * same millisecond the genrator will sleep for 1 millisecond which will
 87:      * reset the salt and allow for another {@link #MAX_SALT} timestamps to be
 88:      * generated.
 89:      * @return a sorted timestamp that is unique within the scope of this
 90:      *         UniqueTimestampGenerator instance
 91:      */
 92:     public long next() {
 93:         long sysTime = System.currentTimeMillis();
 94: 
 95:         updateSalt(sysTime);
 96: 
 97:         // If we reached the maximum salt value wait 1ms
 98:         // (this will reset the salt)
 99:         if (salt > MAX_SALT) {
100:             try {
101:                 Thread.sleep(1);
102:             } catch (InterruptedException e) {
103:                 // Ignore
104:             }
105:             return next();
106:         }
107: 
108:         return applySalt(sysTime);
109:     }
110: 
111:     /**
112:      * WARNING: Debug only - this method throws a RuntimeException if it is
113:      * impossible to create a unique timestamp.
114:      * @param systemTime The system time.
115:      * @return a unique long representing the system time.
116:      */
117:     long next(long systemTime) {
118:         updateSalt(systemTime);
119: 
120:         if (salt > MAX_SALT) {
121:             // We are screwed since this method can not just sleep() to wait for
122:             // a new system time...
123:             throw new RuntimeException("Forced to create non-unique salted timestamp");
124:         }
125: 
126:         return applySalt(systemTime);
127:     }
128: 
129:     /**
130:      * Applies the salt to the given system time. The returned value is valid
131:      * timestamp. To re-create the system time use {@link #systemTime(long)}
132:      *
133:      * @param systemTime the system timestamp to apply the salt to
134:      * @return a valid timestamp in the context of this class - ie. not a system
135:      *         time, but a salted timestamp that is a globally unique identifier
136:      */
137:     private long applySalt(long systemTime) {
138:         systemTime = systemTime << SALT_BITS;
139:         systemTime = systemTime | salt;
140:         return systemTime;
141:     }
142: 
143:     /**
144:      * If the system time has changed reset the salt,
145:      * the stamp will still be unique because the time-part has changed
146:      * @param systemTime the system time that should be used to calculate the
147:      *                   next salt value for
148:      */
149:     private void updateSalt(long systemTime) {
150:         if (systemTime == last) {
151:             salt++;
152:         } else {
153:             salt = 0;
154:             last = systemTime;
155:         }
156:     }
157: 
158:     /**
159:      * Extract the system time from a timestamp generated with {@link #next()}.
160:      * @param timestamp the system time as returned by
161:      * {@link System#currentTimeMillis()} on the creation of {@code timeStamp}
162:      * @return a unique time stamp.
163:      */
164:     public long systemTime(long timestamp) {
165:         return timestamp >>> SALT_BITS;
166:     }
167: 
168:     /**
169:      * Extract the salt part of {@code timeStamp}. The salt is unique for each
170:      * millisecond. That is for two time stamps {@code t1, t2} where
171:      * {@code systemTime(t1) == systemTime(t2)} the salt is guaranteed to be
172:      * different
173:      * @param timestamp the timestamp as generated by {@link #next()} to extract
174:      *                  the salt value from
175:      * @return the salt applied to timestamp to make it unique
176:      */
177:     public long salt(long timestamp) {
178:         // Clear the first TIME_BITS and return
179:         // (only the salt should be untouched)
180:         timestamp = timestamp << TIME_BITS;
181:         return timestamp >>> TIME_BITS;
182:     }
183: 
184:     /**
185:      * Return a salted timestamp that corresponds to the first unique timestamp
186:      * that would be created for {@code systemTime}. Use this method when you
187:      * need to check a system time against a salted timestamp.
188:      *
189:      * @param systemTime system time to convert to a salted timestamp
190:      *                   representing the first timestamp that would be
191:      *                   generated for {@code systemTime}
192:      * @return a timestamp useful for comparison against other timestamps
193:      */
194:     public long baseTimestamp (long systemTime) {
195:         return systemTime << SALT_BITS;
196:     }
197: 
198:     /**
199:      * Return a string representation of the salted timestamp {@code timestamp}.
200:      * The string is formatted as {@code "yyyy-MM-dd'T'HH:mm:ss.SSS"}, eg.
201:      * {@code 2001-07-04T12:08:56.235}.
202:      *
203:      * @param timestamp the salted timestamp to format
204:      * @return string formatted timestamp as described above
205:      */
206:     public String formatTimestamp(long timestamp){
207:         return formatSystemTime(systemTime(timestamp));
208:     }
209: 
210:     /**
211:      * Return a string representation of the (unsalted) system time
212:      * {@code systemTime} (as obtained from {@code System.currentTimeMillis()}).
213:      * The string is formatted as {@code "yyyy-MM-dd'T'HH:mm:ss.SSS"}, eg.
214:      * {@code 2001-07-04T12:08:56.235}.
215:      *
216:      * @param systemTime the system time, as obtained from
217:      *                   {@code System.currentTimeMillis()}
218:      * @return string formatted timestamp as described above
219:      */
220:     public String formatSystemTime(long systemTime) {
221:         return dateFormat.format(new Date(systemTime));
222:     }
223: 
224:     /**
225:      * Return a (unsalted) system time in milliseconds parsed from the string
226:      * {@code systemTime} based on the pattern
227:      * {@code "yyyy-MM-dd'T'HH:mm:ss.SSS"}.
228:      * <p/>
229:      * Use the method {@link #baseTimestamp(long)} to convert the result
230:      * into a salted timestamp. But beware that a timestamp generated this
231:      * way may not be unique.
232:      * @param systemTime string formatted system time as described above
233:      * @return number of milliseconds since the Unix Epoch
234:      * @throws ParseException if unable to parse String. 
235:      */
236:     public long parseSystemTime(String systemTime) throws ParseException {
237:         return dateFormat.parse(systemTime).getTime();
238:     }
