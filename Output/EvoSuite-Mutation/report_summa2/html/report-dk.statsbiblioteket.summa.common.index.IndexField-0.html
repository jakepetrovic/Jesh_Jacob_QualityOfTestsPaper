 32: /**
 33:  * IndexFields are used on several levels. From abstract to concrete, their
 34:  * roles are as follows:
 35:  * </p><p>
 36:  * - Type role: The field defines how to handle content.<br />
 37:  * - Named field role: The field has a concrete name.<br />
 38:  * - Instantiated field: The field has content and is ready for indexing.
 39:  * </p><p>
 40:  * The class is generic. In order to use it properly, the A (Analyzer) and
 41:  * F (Filter) must be specified. It is also strongly recommended to override
 42:  * the methods
 43:  * {@link #analyzerToXMLFragment(Object)},
 44:  * {@link #createAnalyzer(Node)},
 45:  * {@link #tokenizerToXMLFragment(Object)},
 46:  * {@link #createTokenizer(Node)},
 47:  * {@link #filterToXMLFragment(Object)} and
 48:  * {@link #createFilter(Node)},
 49:  * to conform to the standards of the chosen
 50:  * index-format.
 51:  * </p><p>
 52:  * The IndexField is inspired heavily by Lucene Fields and partly by the SOLR
 53:  * schema.
 54:  */

 56:         state = QAInfo.State.IN_DEVELOPMENT,
 57:         author = "te")

 62:      * The boost used when no explicit boost is specified.
 63:      */
 64:     public final static Float DEFAULT_BOOST = 1.0F;
 65: 
 66:     /**
 67:      * If the field is to be used as a sort field, it can either be lazy or
 68:      * active. If active, the sort-structure is generated upon index open.
 69:      * </p><p>
 70:      * Default is lazy.
 71:      */
 72:     public enum SORT_CACHE {
 73:         lazy, active;
 74: 
 75:         public static SORT_CACHE defaultCache() {
 76:             return lazy;
 77:         }
 78: 
 79:         public static SORT_CACHE parse(String cache) {
 80:             if (active.toString().equals(cache)) {
 81:                 return active;
 82:             }
 83:             return lazy;
 84:         }
 85:     }
 86: 
 87:     /**
 88:      * The name of the field is used verbatim as the field in generated indexes.
 89:      * </p><p>
 90:      * This is used at index- and query-time.
 91:      */
 92:     private String name = "Summa-class-default";
 93: 
 94:     /**
 95:      * The name of the default-field - this is used if no parent is specified.

 97:     public static final String SUMMA_DEFAULT = "summa_default";

508:     private void parseAnalyzer(Node node) throws ParseException {
509:         //String typeAttr = ParseUtil.getValue(xPath, node, "@type",
510:         //                                     (String)null);
511:         //String typeAttr = DOM.selectString(node, "@type", null);
512:         String typeAttr = getString(node, "type", null);
513:         boolean parseIndex = false;
514:         boolean parseQuery = false;
515:         //noinspection DuplicateStringLiteralInspection
516:         if ("index".equals(typeAttr)) {
517:             log.trace("Parsing analyzer information for index");
518:             parseIndex = true;
519:         } else if ("query".equals(typeAttr)) {
520:             log.trace("Parsing analyzer information for query");
521:             parseQuery = true;
522:         } else if (typeAttr == null) {
523:             log.trace("Parsing analyzer information for index and query");
524:             parseIndex = true;
525:             parseQuery = true;
526:         } else {
527:             log.warn("Unknown value for attribute type in element analyzer: '"
528:                     + typeAttr + "'. Assigning setup to both index and query");
529:             parseIndex = true;
530:             parseQuery = true;
531:         }
532: 
533:         A analyzer = createAnalyzer(node);
534:         T tokenizer = null;
535:         List<F> filters = null;
536:         try {
537:             String TOKENIZER = "tokenizer";
538:             if ((Boolean) xPath.evaluate(TOKENIZER, node, XPathConstants.BOOLEAN)) {
539:                 tokenizer = createTokenizer((Node) xPath.evaluate(TOKENIZER, node, XPathConstants.BOOLEAN));
540:             }
541:             String FILTER = "filter";
542:             if ((Boolean) xPath.evaluate(FILTER, node, XPathConstants.BOOLEAN)) {
543:                 NodeList filterNodes = (NodeList) xPath.evaluate(FILTER, node, XPathConstants.NODESET);
544:                 filters = new ArrayList<F>(filterNodes.getLength());
545:                 for (int i = 0; i < filterNodes.getLength(); i++) {
546:                     F filter = createFilter(filterNodes.item(i));
547:                     if (filter != null) {
548:                         filters.add(filter);
549:                     }
550:                 }
551:             }
552:         } catch (XPathExpressionException e) {
553:             throw (ParseException) new ParseException(
554:                     "Error evaluating expression 'tokenizer'", -1).initCause(e);
555:         }
556:         if (parseIndex) {
557:             indexAnalyzer = analyzer == null ? indexAnalyzer : analyzer;
558:             indexTokenizer = tokenizer == null ? indexTokenizer : tokenizer;
559:             indexFilters = filters == null ? indexFilters : filters;
560:         }
561:         if (parseQuery) {
562:             queryAnalyzer = analyzer == null ? queryAnalyzer : analyzer;
563:             queryTokenizer = tokenizer == null ? queryTokenizer : tokenizer;
564:             queryFilters = filters == null ? queryFilters : filters;
565:         }
566:     }

733:                 && equals(queryBoost, other.getQueryBoost(), 0.001f)
734:                 && equals(indexBoost, other.getIndexBoost(), 0.001f)
735:                 && nullCompare(sortLocale, other.getSortLocale())
736:                 && sortCache == other.getSortCache()
737:                 && inFreetext == other.isInFreetext()
738:                 && required == other.isRequired()
