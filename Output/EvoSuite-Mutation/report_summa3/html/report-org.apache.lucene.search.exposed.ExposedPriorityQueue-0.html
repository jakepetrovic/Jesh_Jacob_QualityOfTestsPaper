 20: import org.apache.lucene.search.exposed.compare.ComparatorFactory;
 21: 
 22: /**
 23:  * A PriorityQueue maintains a partial ordering of its elements such that the
 24:  * least element can always be found in constant time.  Put()'s and pop()'s
 25:  * require log(size) time.
 26:  *
 27:  * <p><b>NOTE</b>: This class pre-allocates a full array of
 28:  * length <code>maxSize+1</code>, in {@link #initialize}.
 29:  *
 30:  * @lucene.internal
 31:  *
 32:  * Originally taken from PriorityQueue.
 33:  * Modified to use int[] and to take a custom comparator for memory efficiency
 34: */
 35: public class ExposedPriorityQueue {
 36:   private int size;
 37:   private int maxSize;
 38:   protected int[] heap;
 39:   private ComparatorFactory.OrdinalComparator comparator;
 40: 
 41:   public ExposedPriorityQueue(
 42:       ComparatorFactory.OrdinalComparator comparator, int size) {
 43:     this.comparator = comparator;
 44:     initialize(size);
 45:   }
 46: 
 47:   protected boolean lessThan(int a, int b) {
 48:       return comparator.compare(a, b) < 0;
 49:   }
 50: 
 51:   /**
 52:    * This method can be overridden by extending classes to return a sentinel
 53:    * object which will be used by {@link #initialize(int)} to fill the queue, so
 54:    * that the code which uses that queue can always assume it's full and only
 55:    * change the top without attempting to insert any new object.<br>
 56:    *
 57:    * Those sentinel values should always compare worse than any non-sentinel
 58:    * value (i.e., {@link #lessThan} should always favor the
 59:    * non-sentinel values).<br>
 60:    *
 61:    * By default, this method returns false, which means the queue will not be
 62:    * filled with sentinel values. Otherwise, the value returned will be used to
 63:    * pre-populate the queue. Adds sentinel values to the queue.<br>
 64:    *
 65:    * If this method is extended to return a non-null value, then the following
 66:    * usage pattern is recommended:
 67:    *
 68:    * <pre>
 69:    * // extends getSentinelObject() to return a non-null value.
 70:    * PriorityQueue<MyObject> pq = new MyQueue<MyObject>(numHits);
 71:    * // save the 'top' element, which is guaranteed to not be null.
 72:    * MyObject pqTop = pq.top();
 73:    * &lt;...&gt;
 74:    * // now in order to add a new element, which is 'better' than top (after
 75:    * // you've verified it is better), it is as simple as:
 76:    * pqTop.change().
 77:    * pqTop = pq.updateTop();
 78:    * </pre>
 79:    *
 80:    * <b>NOTE:</b> if this method returns a non-null value, it will be called by
 81:    * {@link #initialize(int)} {@link #size()} times, relying on a new object to
 82:    * be returned and will not check if it's null again. Therefore you should
 83:    * ensure any call to this method creates a new instance and behaves
 84:    * consistently, e.g., it cannot return null if it previously returned
 85:    * non-null.
 86:    *
 87:    * @return the sentinel object to use to pre-populate the queue, or null if
 88:    *         sentinel objects are not supported.
 89:    */
 90:   protected int getSentinelObject() {
 91:     return -1;
 92:   }
 93: 
 94:   /** Subclass constructors must call this. */
 95:   @SuppressWarnings("unchecked")
 96:   protected final void initialize(int maxSize) {
 97:     size = 0;
 98:     int heapSize;
 99:     if (0 == maxSize)
100:       // We allocate 1 extra to avoid if statement in top()
101:       heapSize = 2;
102:     else {
103:       if (maxSize == Integer.MAX_VALUE) {
104:         // Don't wrap heapSize to -1, in this case, which
105:         // causes a confusing NegativeArraySizeException.
106:         // Note that very likely this will simply then hit
107:         // an OOME, but at least that's more indicative to
108:         // caller that this values is too big.  We don't +1
109:         // in this case, but it's very unlikely in practice
110:         // one will actually insert this many objects into
111:         // the PQ:
112:         heapSize = Integer.MAX_VALUE;
113:       } else {
114:         // NOTE: we add +1 because all access to heap is
115:         // 1-based not 0-based.  heap[0] is unused.
116:         heapSize = maxSize + 1;
117:       }
118:     }
119:     heap = new int[heapSize];
120:     this.maxSize = maxSize;
121: 
122:     // If sentinel objects are supported, populate the queue with them
123:     int sentinel = getSentinelObject();
124:     if (sentinel != -1) {
125:       heap[1] = sentinel;
126:       for (int i = 2; i < heap.length; i++) {
127:         heap[i] = getSentinelObject();
128:       }
129:       size = maxSize;
130:     }
131:   }
132: 
133:   /**
134:    * Adds an Object to a PriorityQueue in log(size) time. If one tries to add
135:    * more objects than maxSize from initialize an
136:    * {@link ArrayIndexOutOfBoundsException} is thrown.
137:    *
138:    * @return the new 'top' element in the queue.
139:    */
140:   public final int add(int element) {
141:     size++;
142:     heap[size] = element;
143:     upHeap();
144:     return heap[1];
145:   }
146: 
147:   /**
148:    * Adds an Object to a PriorityQueue in log(size) time.
149:    * It returns the object (if any) that was
150:    * dropped off the heap because it was full. This can be
151:    * the given parameter (in case it is smaller than the
152:    * full heap's minimum, and couldn't be added), or another
153:    * object that was previously the smallest value in the
154:    * heap and now has been replaced by a larger one, or null
155:    * if the queue wasn't yet full with maxSize elements.
156:    */
157:   public int insertWithOverflow(int element) {
158:     if (size < maxSize) {
159:       add(element);
160:       return -1;
161:     } else if (size > 0 && !lessThan(element, heap[1])) {
162:       int ret = heap[1];
163:       heap[1] = element;
164:       updateTop();
165:       return ret;
166:     } else {
167:       return element;
168:     }
169:   }
170: 
171:   /** Returns the least element of the PriorityQueue in constant time. */
172:   public final int top() {
173:     // We don't need to check size here: if maxSize is 0,
174:     // then heap is length 2 array with both entries null.
175:     // If size is 0 then heap[1] is already null.
176:     return heap[1];
177:   }
178: 
179:   /** Removes and returns the least element of the PriorityQueue in log(size)
180:     time. */
181:   public final int pop() {
182:     if (size > 0) {
183:       int result = heap[1];			  // save first value
184:       heap[1] = heap[size];			  // move last to first
185:       heap[size] = -1;			  // permit GC of objects
186:       size--;
187:       downHeap();				  // adjust heap
188:       return result;
189:     } else
190:       return -1;
191:   }
192: 
193:   /**
194:    * Should be called when the Object at top changes values. Still log(n) worst
195:    * case, but it's at least twice as fast to
196:    *
197:    * <pre>
198:    * pq.top().change();
199:    * pq.updateTop();
200:    * </pre>
201:    *
202:    * instead of
203:    *
204:    * <pre>
205:    * o = pq.pop();
206:    * o.change();
207:    * pq.push(o);
208:    * </pre>
209:    *
210:    * @return the new 'top' element.
211:    */
212:   public final int updateTop() {
213:     downHeap();
214:     return heap[1];
215:   }
216: 
217:   /** Returns the number of elements currently stored in the PriorityQueue. */
218:   public final int size() {
219:     return size;
220:   }
221: 
222:   /** Removes all entries from the PriorityQueue. */
223:   public final void clear() {
224:     for (int i = 0; i <= size; i++) {
225:       heap[i] = -1;
226:     }
227:     size = 0;
228:   }
229: 
230:   private final void upHeap() {
231:     int i = size;
232:     int node = heap[i];			  // save bottom node
233:     int j = i >>> 1;
234:     while (j > 0 && lessThan(node, heap[j])) {
235:       heap[i] = heap[j];			  // shift parents down
236:       i = j;
237:       j = j >>> 1;
238:     }
239:     heap[i] = node;				  // install saved node
240:   }
241: 
242:   private final void downHeap() {
243:     int i = 1;
244:     int node = heap[i];			  // save top node
245:     int j = i << 1;				  // find smaller child
246:     int k = j + 1;
247:     if (k <= size && lessThan(heap[k], heap[j])) {
248:       j = k;
249:     }
250:     while (j <= size && lessThan(heap[j], node)) {
251:       heap[i] = heap[j];			  // shift up child
252:       i = j;
253:       j = i << 1;
254:       k = j + 1;
255:       if (k <= size && lessThan(heap[k], heap[j])) {
256:         j = k;
257:       }
258:     }
259:     heap[i] = node;				  // install saved node
260:   }
261: }
