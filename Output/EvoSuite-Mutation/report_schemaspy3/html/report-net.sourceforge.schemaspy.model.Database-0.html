162:         private final String clazz;
163:         private final Pattern include;
164:         private final Pattern exclude;

167:         /**
168:          * @param clazz table or view
169:          * @param include
170:          * @param exclude
171:          * @param verbose
172:          * @param validTypes
173:          */
174:         NameValidator(String clazz, Pattern include, Pattern exclude, String[] validTypes) {
175:             this.clazz = clazz;
176:             this.include = include;
177:             this.exclude = exclude;
178:             this.validTypes = new HashSet<String>();
179:             for (String type : validTypes)
180:             {
181:                 this.validTypes.add(type.toUpperCase());
182:             }
183:         }
184: 
185:         /**
186:          * Returns <code>true</code> if the table/view name is deemed "valid"
187:          *
188:          * @param name name of the table or view
189:          * @param type type as returned by metadata.getTables():TABLE_TYPE
190:          * @return
191:          */
192:         boolean isValid(String name, String type) {
193:             // some databases (MySQL) return more than we wanted
194:             if (!validTypes.contains(type.toUpperCase()))
195:                 return false;
196: 
197:             // Oracle 10g introduced problematic flashback tables
198:             // with bizarre illegal names
199:             if (name.indexOf("$") != -1) {
200:                 if (fineEnabled) {
201:                     logger.fine("Excluding " + clazz + " " + name +
202:                                 ": embedded $ implies illegal name");
203:                 }
204:                 return false;
205:             }
206: 
207:             if (exclude.matcher(name).matches()) {
208:                 if (fineEnabled) {
209:                     logger.fine("Excluding " + clazz + " " + name +
210:                                 ": matches exclusion pattern \"" + exclude + '"');
211:                 }
212:                 return false;
213:             }
214: 
215:             boolean valid = include.matcher(name).matches();
216:             if (fineEnabled) {
217:                 if (valid) {
218:                     logger.fine("Including " + clazz + " " + name +
219:                                 ": matches inclusion pattern \"" + include + '"');
220:                 } else {
221:                     logger.fine("Excluding " + clazz + " " + name +
222:                                 ": doesn't match inclusion pattern \"" + include + '"');
223:                 }
224:             }
225:             return valid;
226:         }
227:     }

236:         final Pattern include = config.getTableInclusions();
237:         final Pattern exclude = config.getTableExclusions();
238:         final int maxThreads = config.getMaxDbThreads();
239: 
240:         String[] types = getTypes("tableTypes", "TABLE");
241:         NameValidator validator = new NameValidator("table", include, exclude, types);
242:         List<BasicTableMeta> entries = getBasicTableMeta(metadata, true, types);
243: 
244:         TableCreator creator;
245:         if (maxThreads == 1) {
246:             creator = new TableCreator();
247:         } else {
248:             // creating tables takes a LONG time (based on JProbe analysis),
249:             // so attempt to speed it up by doing several in parallel.
250:             // note that it's actually DatabaseMetaData.getIndexInfo() that's expensive
251: 
252:             creator = new ThreadedTableCreator(maxThreads);
253: 
254:             // "prime the pump" so if there's a database problem we'll probably see it now
255:             // and not in a secondary thread
256:             while (!entries.isEmpty()) {
257:                 BasicTableMeta entry = entries.remove(0);
258: 
259:                 if (validator.isValid(entry.name, entry.type)) {
260:                     new TableCreator().create(entry);
261:                     break;
262:                 }
263:             }
264:         }
265: 
266:         // kick off the secondary threads to do the creation in parallel
267:         for (BasicTableMeta entry : entries) {
268:             if (validator.isValid(entry.name, entry.type)) {
269:                 creator.create(entry);
270:             }
271:         }
272: 
273:         // wait for everyone to finish
274:         creator.join();
275:     }
