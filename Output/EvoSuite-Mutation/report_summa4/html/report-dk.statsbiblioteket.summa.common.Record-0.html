 66:          *
 67:          * @param in The string representation of the wanted state.

 76:                 return invalid;
 77:             } else {
 78:                 log.warn("Unknown state: '" + in + "' in ValidationState.fromString. Returning " + notValidated);
 79:                 return notValidated;
 80:             }
 81:         }
