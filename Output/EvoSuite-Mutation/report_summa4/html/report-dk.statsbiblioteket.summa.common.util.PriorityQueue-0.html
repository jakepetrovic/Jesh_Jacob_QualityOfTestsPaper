 26: /**
 27:  * A generic Priority Queue. This Queue supports a max capacity.
 28:  * If an element is inserted and the max capacity is exceeded, the least
 29:  * significant value is discarded and returned.
 30:  * The queue optionally takes a Comparator and uses the natural order from
 31:  * the Comparable objects if none is specified.
 32:  * insert:    O(log(n))
 33:  * removeMin: O(log(n))
 34:  * getMin:    O(1)
 35:  * clear:     O(1)
 36:  * Note: This queue is not thread-safe.
 37:  *
 38:  * see http://en.wikibooks.org/wiki/Wikiversity:Data_Structures#Heaps
 39:  * and http://www.personal.kent.edu/~rmuhamma/Algorithms/MyAlgorithms/Sorting/heapSort.htm
 40:  * </p><p>
 41:  * Note: This class creates arrays of generics by using the same cast-hack as
 42:  *       {@link ArrayList}.
 43:  *       See http://forums.sun.com/thread.jspa?threadID=564355&forumID=316 for
 44:  *       a discussion on the subject.
 45:  */
 46: @QAInfo(level = QAInfo.Level.NORMAL,
 47:         state = QAInfo.State.IN_DEVELOPMENT,
 48:         author = "te")
 49: @SuppressWarnings({"unchecked"})
 50: public class PriorityQueue<T extends Comparable<? super T>> {
 51:     private T[] heap;
 52:     private int size = 0;
 53: 
 54:     private int maxCapacity;
 55:     private static final int DEFAULT_INITIAL_CAPACITY = 1000;
 56:     private final Comparator<T> comparator; // Optional
 57: 
 58:     public PriorityQueue() {
 59:         this(Integer.MAX_VALUE);
 60:     }
 61:     public PriorityQueue(int maxCapacity) {
 62:         this(null, Math.min(DEFAULT_INITIAL_CAPACITY, maxCapacity), maxCapacity);
 63:     }
 64:     public PriorityQueue(Comparator<T> comparator, int maxCapacity) {
 65:         this(comparator, Math.min(DEFAULT_INITIAL_CAPACITY, maxCapacity), maxCapacity);
 66:     }
 67:     public PriorityQueue(int initialCapacity, int maxCapacity) {
 68:         this(null, initialCapacity, maxCapacity);
 69:     }
 70:     public PriorityQueue(Comparator<T> comparator, int initialCapacity, int maxCapacity) {
 71:         if (maxCapacity < initialCapacity) {
 72:             throw new IllegalArgumentException(String.format(
 73:                     "The max capacity (%d) must be larger than or equal to the initial capacity (%d)",
 74:                     maxCapacity, initialCapacity));
 75:         }
 76:         //noinspection unchecked
 77:         heap = (T[])new Comparable[initialCapacity];
 78:         this.maxCapacity = maxCapacity;
 79:         this.comparator = fixComparator(comparator);
 80:     }
 81: 
 82:     // Ensures that there is a Comparator (defaults to narutal order)
 83:     private Comparator<T> fixComparator(Comparator<T> comparator) {
 84:         if (comparator != null) {
 85:             return comparator;
 86:         }
 87:         return new Comparator<T>() {
 88:             @Override
 89:             public final int compare(final T o1, final T o2) {
 90:                 return o1.compareTo(o2);
 91:             }
 92:         };
 93:     }
 94: 
 95:     /**
 96:      * Assign all the given values to the queue, replacing all existing values.
 97:      * Time complexity is O(n).
 98:      * @param values      the values to assign to the queue.
 99:      * @param size        the number of values to assign. This must be equal to
100:      *                    or smaller than the number of values.
101:      * @param reuseArray  if true, the array with the values is used for the
102:      *                    internal heap-representation.
103:      * @param maxCapacity the maximum capacity for this queue. The maximum must
104:      *                    be equal to or greater than the size.
105:      */
106:     public void setValues(T[] values, int size, boolean reuseArray,
107:                           int maxCapacity) {
108:         if (size > values.length) {
109:             throw new IllegalArgumentException(String.format(
110:                     "The size (%d) must be equal to or smaller than the length of values (%d)",
111:                     size, values.length));
112:         }
113:         if (maxCapacity < size) {
114:             throw new IllegalArgumentException(String.format(
115:                     "The maxCapacity (%d) must be equal to or greater than size (%d)",
116:                     maxCapacity, size));
117:         }
118:         this.maxCapacity = maxCapacity;
119:         this.size = size;
120:         if (reuseArray) {
121:             heap = values;
122:         } else {
123:             //noinspection unchecked
124:             heap = (T[])new Comparable[Math.max(Math.min(DEFAULT_INITIAL_CAPACITY, maxCapacity), size)];
125:             System.arraycopy(values, 0, heap, 0, size);
126:         }
127: 
128:         // This is O(n), although it looks like O(n*log(n))
129:         for (int position = size / 2 - 1 ; position  >= 0 ; position --) {
130:             siftDown(position);
131:         }
132:     }
133: 
134:     /**
135:      * Insert a value in the queue. Time complexity is O(log(n)).
136:      * If the insertion pushed out an existing value, that value is returned.
137:      * @param value the value to insert in the queue.
138:      * @return the old value, if the new value pushes one out. Else null
139:      */
140:     public T insert(final T value) {
141:         if (size == maxCapacity) { // Heap is full
142:             //noinspection unchecked
143:             final T heapZero = heap[0];
144:             if (comparator.compare(heapZero, value) > 0) {
145:                 heap[0] = value;
146:                 siftDown();
147:                 return heapZero;
148:             }
149:         } else { // Insert at the end and sift up
150:             if (heap.length == size+1) { // Expand heap
151:                 int newSize = Math.min(maxCapacity, heap.length*2);
152:                 //noinspection unchecked
153:                 T[] newQueue = (T[])new Comparable[newSize];
154:                 System.arraycopy(heap, 0, newQueue, 0, size);
155:                 heap = newQueue;
156:             }
157:             heap[size++] = value;
158:             siftUp();
159:         }
160:         return null;
161:     }
162: 
163:     /**
164:      * Remove the smallest value on the queue Time complexity is O(log(n)).
165:      * @return the smallest value on the queue.
166:      */
167:     public T removeMin() {
168:         final T result = getMin();
169:         heap[0] = heap[size-1];
170:         size--;
171:         siftDown();
172:         return result;
173:     }
174: 
175:     /**
176:      * Get the smallest value on the queue. Time complexity is O(1).
177:      * @return the smallest value on the queue.
178:      */
179:     public T getMin() {
180:         if (size == 0) {
181:             throw new ArrayIndexOutOfBoundsException("No values left on the heap");
182:         }
183:         //noinspection unchecked
184:         return heap[0];
185:     }
186: 
187:     /**
188:      * @return the size of the queue.
189:      */
190:     public int getSize() {
191:         return size;
192:     }
193: 
194:     /**
195:      * Assumes that the last element in the queue is positioned wrong and
196:      * balances the heap accordingly.
197:      */
198:     protected void siftUp() {
199:         int position = size-1;
200:         while (position > 0) {
201:             final int parentPosition = parent(position);
202:             //noinspection unchecked
203:             if (comparator.compare(heap[parentPosition], heap[position]) < 0) {
204:                 swap(parentPosition, position);
205:             } else {
206:                 break;
207:             }
208:             position = parentPosition;
209:         }
210:     }
211: 
212:     /**
213:      * Assumes that the first element (the root) in the queue is positioned
214:      * wrong and balances the heap accordingly.
215:      */
216:     protected void siftDown() {
217:         siftDown(0);
218:     }
219:     protected void siftDown(final int startPosition) {
220:         int position = startPosition;
221:         while (firstChild(position) < size) {
222:             int kid = firstChild(position);
223:             //noinspection unchecked
224:             if (kid < size-1 &&
225:                 comparator.compare(heap[kid], heap[kid+1]) < 0) {
226:                 kid++;
227:             }
228:             //noinspection unchecked
229:             if (comparator.compare(heap[position], heap[kid]) > 0) {
230:                 break;
231:             } else {
232:                 swap(kid, position);
233:                 position = kid;
234:             }
235:         }
236:     }
237: 
238:     /**
239:      * @param element the position of an element.
240:      * @return the position of the parent of the element.
241:      */
242:     protected int parent(int element) {
243:       return (element-1) / 2;
244:     }
245: 
246:     /**
247:      * @param element the position of an element.
248:      * @return the position of the first child of the element. Note that this
249:      *         might be outside of the heap.
250:      */
251:     protected int firstChild(int element) {
252:         return 2*element+1;
253:     }
254: 
255:     private void swap(int element1, int element2) {
256:         final T temp = heap[element1];
257:         heap[element1] = heap[element2];
258:         heap[element2] = temp;
259:     }
260: 
261:     /**
262:      * @return the comparator used for this queue. If no explicit Comparator is
263:      * given on creation time, this will use natural ordering.
264:      */
265:     public Comparator<T> getComparator() {
266:         return comparator;
267:     }
268: 
269:     /**
270:      * Pretty print the heap.
271:      * @return string with all element and there placement in internal heap.
272:      */
273:     public String toString() {
274:         String str = "{";
275:         for(int i=0; i<size; i++) {
276:             str += heap[i] + ",";
277:         }
278:         return str + "}\n";
279:     }
280: }
