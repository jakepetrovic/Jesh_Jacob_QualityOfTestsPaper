 58: /**
 59:  * Serialize SAX2 events to its textual XML representation.
 60:  *<p>
 61:  * Support output to XML, XHTML, HTML and Text. Full support for XML namespaces.
 62:  *<p>
 63:  * The {@link #startDocument} and {@link #endDocument} method must be used.
 64:  *<p>
 65:  * Output properties controls the serialization process, it uses the keys in 
 66:  * <code>javax.xml.transform.OutputKeys</code>.
 67:  *<p>
 68:  * The METHOD output property must be specified. It can be 
 69:  * "xml", "xhtml", "text" or "html".
 70:  *<p>
 71:  * The CDATA_SECTION_ELEMENTS output property is not supported.
 72:  *<p>
 73:  * The ENCODING output property defaults to UTF-8 for XML and XHTML, 
 74:  * and to ISO-8859-1 for Text and HTML.
 75:  *<p>
 76:  * The "html" output method does <em>not</em> escape non-ASCII characters 
 77:  * in URI attribute values, as specified in the XSLT 1.0 specification.
 78:  *<p>
 79:  * <code>javax.xml.transform.Result.PI_DISABLE_OUTPUT_ESCAPING</code> 
 80:  * and 
 81:  * <code>javax.xml.transform.Result.PI_ENABLE_OUTPUT_ESCAPING</code>
 82:  *  can be used as processingInstruction targets to disable output escaping. 
 83:  */

 85:                                             DTDHandler, DeclHandler

 88:     protected final String systemId;
 89:     protected final XMLCharacterEncoder out;

 94:     /**
 95:      * Factory method, create a new Serializer.
 96:      *
 97:      * @param result  specifies where to write the textual representation.
 98:      * @param outputProperties output properties, uses {@link javax.xml.transform.OutputKeys}
 99:      *
100:      * @return a new Serializer
101:      *
102:      * @throws UnsupportedEncodingException if the ENCODING output property 
103:      *         is invalid.
104:      * @throws IllegalArgumentException if any other output property is invalid.
105:      * @throws IOException if the result is invalid.
106:      */
107:     public static Serializer createSerializer(
108:             StreamResult result, Properties outputProperties)
109:         throws IllegalArgumentException, IOException, 
110:             UnsupportedEncodingException
111:     {
112:         OutputConfig outputConfig = 
113:             OutputConfig.createOutputConfig(outputProperties); 
114:         
115:         if (outputConfig.isHtml)
116:             return new HTMLSerializer(result, outputConfig);
117:         else if (outputConfig.isXhtml)
118:             return new XMLSerializer(result, outputConfig);
119:         else if (outputConfig.isText)
120:             return new TextSerializer(result, outputConfig);
121:         else // XML
122:             return new XMLSerializer(result, outputConfig);
123:     }
124: 
125:     
126:     protected Serializer(StreamResult result, OutputConfig outputConfig)
127:         throws IllegalArgumentException, IOException, 
128:             UnsupportedEncodingException
129:     {
130:         this.outputConfig = outputConfig;
131:         
132:         if (!outputConfig.cdata_section_elements.isEmpty())
133:         {
134:             throw new IllegalArgumentException(
135:                 "cdata_section_elements is not supported");
136:         }
137: 
138:         this.systemId = result.getSystemId();
139:         OutputStream os = result.getOutputStream();
140:         Writer w = result.getWriter();
141:         if (os != null)
142:         {
143:             out = new XMLCharacterEncoder(os, outputConfig.encoding);
144:             doClose = false;
145:         }
146:         else if (w != null)
147:         {
148:             out = new XMLCharacterEncoder(w);
149:             doClose = false;
150:         }
151:         else if (systemId != null)
152:         {
153:             OutputStream _os;
154:             
155:             try {
156:                 URI uri = new URI(systemId);    
157:                 if (!uri.isAbsolute())
158:                 {
159:                     File file = new File(systemId);
160:                     _os = new FileOutputStream(file);
161:                 }
162:                 else if (uri.getScheme().equals("file"))
163:                 {
164:                     File file = new File(uri);
165:                     _os = new FileOutputStream(file);
166:                 }
167:                 else // Absolute URI with other scheme than "file:"
168:                 {
169:                     URL url = uri.toURL();
170:                     URLConnection urlConn = url.openConnection();
171:                     urlConn.setDoOutput(true);
172:                     _os = urlConn.getOutputStream();
173:                 }
174:             }
175:             catch (URISyntaxException e)
176:             {
177:                 throw new IllegalArgumentException(
178:                     "Invalid systemId: " + e.getMessage());    
179:             }
180:             
181:             os = new BufferedOutputStream(_os); 
182:             out = new XMLCharacterEncoder(os, outputConfig.encoding);
183:             doClose = true;
184:         }
185:         else
186:         {
187:             throw new IllegalArgumentException("Empty StreamResult");     
188:         }
189:     }
