293:     /**
294:      * Splits the specified delimited String into tokens, supporting quoted tokens so that quoted strings themselves
295:      * won't be tokenized.
296:      *
297:      * <p>This method's implementation is very loosely based (with significant modifications) on
298:      * <a href="http://blogs.bytecode.com.au/glen">Glen Smith</a>'s open-source
299:      * <a href="http://opencsv.svn.sourceforge.net/viewvc/opencsv/trunk/src/au/com/bytecode/opencsv/CSVReader.java?&view=markup">CSVReader.java</a>
300:      * file.
301:      *
302:      * <p>That file is Apache 2.0 licensed as well, making Glen's code a great starting point for us to modify to
303:      * our needs.
304:      *
305:      * @param aLine          the String to parse
306:      * @param delimiter      the delimiter by which the <tt>line</tt> argument is to be split
307:      * @param beginQuoteChar the character signifying the start of quoted text (so the quoted text will not be split)
308:      * @param endQuoteChar   the character signifying the end of quoted text
309:      * @param retainQuotes   if the quotes themselves should be retained when constructing the corresponding token
310:      * @param trimTokens     if leading and trailing whitespace should be trimmed from discovered tokens.
311:      * @return the tokens discovered from parsing the given delimited <tt>line</tt>.
312:      */
313:     public static String[] split(String aLine, char delimiter, char beginQuoteChar, char endQuoteChar,
314:                                  boolean retainQuotes, boolean trimTokens) {
315:         String line = clean(aLine);
316:         if (line == null) {
317:             return null;
318:         }
319: 
320:         List<String> tokens = new ArrayList<String>();
321:         StringBuffer sb = new StringBuffer();
322:         boolean inQuotes = false;
323: 
324:         for (int i = 0; i < line.length(); i++) {
325: 
326:             char c = line.charAt(i);
327:             if (c == beginQuoteChar) {
328:                 // this gets complex... the quote may end a quoted block, or escape another quote.
329:                 // do a 1-char lookahead:
330:                 if (inQuotes  // we are in quotes, therefore there can be escaped quotes in here.
331:                         && line.length() > (i + 1)  // there is indeed another character to check.
332:                         && line.charAt(i + 1) == beginQuoteChar) { // ..and that char. is a quote also.
333:                     // we have two quote chars in a row == one quote char, so consume them both and
334:                     // put one on the token. we do *not* exit the quoted text.
335:                     sb.append(line.charAt(i + 1));
336:                     i++;
337:                 } else {
338:                     inQuotes = !inQuotes;
339:                     if (retainQuotes) {
340:                         sb.append(c);
341:                     }
342:                 }
343:             } else if (c == endQuoteChar) {
344:                 inQuotes = !inQuotes;
345:                 if (retainQuotes) {
346:                     sb.append(c);
347:                 }
348:             } else if (c == delimiter && !inQuotes) {
349:                 String s = sb.toString();
350:                 if (trimTokens) {
351:                     s = s.trim();
352:                 }
353:                 tokens.add(s);
354:                 sb = new StringBuffer(); // start work on next token
355:             } else {
356:                 sb.append(c);
357:             }
358:         }
359:         String s = sb.toString();
360:         if (trimTokens) {
361:             s = s.trim();
362:         }
363:         tokens.add(s);
364:         return tokens.toArray(new String[tokens.size()]);
365:     }
