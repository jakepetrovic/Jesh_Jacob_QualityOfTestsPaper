 24: /**
 25:  * MethodInterceptor that inspects a specific annotation on the method invocation before continuing
 26:  * its execution.
 27:  *
 28:  * @author Les Hazlewood
 29:  * @since 0.9
 30:  */
 31: public abstract class AnnotationMethodInterceptor extends MethodInterceptorSupport {
 32: 
 33:     private AnnotationHandler handler;
 34: 
 35:     /**
 36:      * Constructs an <code>AnnotationMethodInterceptor</code> with the
 37:      * {@link AnnotationHandler AnnotationHandler} that will be used to process annotations of a corresponding
 38:      * type.
 39:      *
 40:      * @param handler the handler to delegate to for processing the annotation.
 41:      */
 42:     public AnnotationMethodInterceptor(AnnotationHandler handler) {
 43:         setHandler(handler);
 44:     }
 45: 
 46:     public AnnotationHandler getHandler() {
 47:         return handler;
 48:     }
 49: 
 50:     public void setHandler(AnnotationHandler handler) {
 51:         this.handler = handler;
 52:     }
 53: 
 54:     /**
 55:      * Returns <code>true</code> if this interceptor supports, that is, should inspect, the specified
 56:      * <code>MethodInvocation</code>, <code>false</code> otherwise.
 57:      * <p/>
 58:      * The default implementation simply does the following:
 59:      * <p/>
 60:      * <code>return {@link #getAnnotation(MethodInvocation) getAnnotation(mi)} != null</code>
 61:      *
 62:      * @param mi the <code>MethodInvocation</code> for the method being invoked.
 63:      * @return <code>true</code> if this interceptor supports, that is, should inspect, the specified
 64:      *         <code>MethodInvocation</code>, <code>false</code> otherwise.
 65:      */
 66:     public boolean supports(MethodInvocation mi) {
 67:         return getAnnotation(mi) != null;
 68:     }
 69: 
 70:     /**
 71:      * Returns the Annotation that this interceptor will process for the specified method invocation.
 72:      * <p/>
 73:      * The default implementation merely gets the underlying {@link Method Method} from the supplied
 74:      * <code>MethodInvocation</code> argument, and returns:
 75:      * <p/>
 76:      * <code>mi.{@link Method#getAnnotation(Class) getAnnotation}({@link AnnotationHandler#getAnnotationClass() handler.getAnnotationClass()});</code>
 77:      *
 78:      * @param mi the MethodInvocation wrapping the Method from which the Annotation will be acquired.
 79:      * @return the Annotation that this interceptor will process for the specified method invocation.
 80:      * @throws IllegalArgumentException if the supplied <code>MethodInvocation</code> argument is <code>null</code> or
 81:      *                                  its underlying <code>Method</code> is <code>null</code>.
 82:      */
 83:     protected Annotation getAnnotation(MethodInvocation mi) throws IllegalArgumentException {
 84:         if (mi == null) {
 85:             throw new IllegalArgumentException("method argument cannot be null");
 86:         }
 87:         Method m = mi.getMethod();
 88:         if (m == null) {
 89:             String msg = MethodInvocation.class.getName() + " parameter incorrectly " +
 90:                     "constructed.  getMethod() returned null";
 91:             throw new IllegalArgumentException(msg);
 92: 
 93:         }
 94:         return m.getAnnotation(getHandler().getAnnotationClass());
 95:     }
 96: 
 97: }
