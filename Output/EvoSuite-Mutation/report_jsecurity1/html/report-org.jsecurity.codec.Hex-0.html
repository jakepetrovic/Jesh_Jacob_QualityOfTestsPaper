 24:  * <p>This class was borrowed from Apache Commons Codec SVN repository (rev. 560660 ) with modifications
 25:  * to enable Hex conversion without a full dependency on Commons Codec.  We didn't want to reinvent the wheel of
 26:  * great work they've done, but also didn't want to force every JSecurity user to depend on the commons-codec.jar</p>

 30:  *
 31:  * @author Apache Software Foundation
 32:  * @author Les Hazlewood
 33:  * @since 0.9
 34:  */

 36: 
 37:     /**
 38:      * Used to build output as Hex
 39:      */
 40:     private static final char[] DIGITS = {
 41:             '0', '1', '2', '3', '4', '5', '6', '7',
 42:             '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'
 43:     };

 45:     /**
 46:      * Encodes the specifed byte array to a character array and then returns that character array
 47:      * as a String.
 48:      *
 49:      * @param bytes the byte array to Hex-encode.
 50:      * @return A String representation of the resultant hex-encoded char array.
 51:      */
 52:     public static String encodeToString(byte[] bytes) {
 53:         char[] encodedChars = encode(bytes);
 54:         return new String(encodedChars);
 55:     }
 56: 
 57:     /**
 58:      * Converts an array of bytes into an array of characters representing the hexidecimal values of each byte in order.
 59:      * The returned array will be double the length of the passed array, as it takes two characters to represent any
 60:      * given byte.
 61:      *
 62:      * @param data byte[] to convert to Hex characters
 63:      * @return A char[] containing hexidecimal characters
 64:      */
 65:     public static char[] encode(byte[] data) {
 66: 
 67:         int l = data.length;
 68: 
 69:         char[] out = new char[l << 1];
 70: 
 71:         // two characters form the hex value.
 72:         for (int i = 0, j = 0; i < l; i++) {
 73:             out[j++] = DIGITS[(0xF0 & data[i]) >>> 4];
 74:             out[j++] = DIGITS[0x0F & data[i]];
 75:         }
 76: 
 77:         return out;
 78:     }
 79: 
 80:     /**
 81:      * Converts an array of character bytes representing hexidecimal values into an
 82:      * array of bytes of those same values. The returned array will be half the
 83:      * length of the passed array, as it takes two characters to represent any
 84:      * given byte. An exception is thrown if the passed char array has an odd
 85:      * number of elements.
 86:      *
 87:      * @param array An array of character bytes containing hexidecimal digits
 88:      * @return A byte array containing binary data decoded from
 89:      *         the supplied byte array (representing characters).
 90:      * @throws IllegalArgumentException Thrown if an odd number of characters is supplied
 91:      *                                  to this function
 92:      * @see #decode(char[])
 93:      */
 94:     public static byte[] decode(byte[] array) throws IllegalArgumentException {
 95:         String s = CodecSupport.toString(array);
 96:         return decode(s);
 97:     }
 98: 
 99:     /**
100:      * Converts the specified Hex-encoded String into a raw byte array.  This is a
101:      * convenience method that merely delegates to {@link #decode(char[])} using the
102:      * argument's hex.toCharArray() value.
103:      *
104:      * @param hex a Hex-encoded String.
105:      * @return A byte array containing binary data decoded from the supplied String's char array.
106:      */
107:     public static byte[] decode(String hex) {
108:         return decode(hex.toCharArray());
109:     }

111:     /**
112:      * Converts an array of characters representing hexidecimal values into an
113:      * array of bytes of those same values. The returned array will be half the
114:      * length of the passed array, as it takes two characters to represent any
115:      * given byte. An exception is thrown if the passed char array has an odd
116:      * number of elements.
117:      *
118:      * @param data An array of characters containing hexidecimal digits
119:      * @return A byte array containing binary data decoded from
120:      *         the supplied char array.
121:      * @throws IllegalArgumentException if an odd number or illegal of characters
122:      *                                  is supplied
123:      */
124:     public static byte[] decode(char[] data) throws IllegalArgumentException {
125: 
126:         int len = data.length;
127: 
128:         if ((len & 0x01) != 0) {
129:             throw new IllegalArgumentException("Odd number of characters.");
130:         }
131: 
132:         byte[] out = new byte[len >> 1];
133: 
134:         // two characters form the hex value.
135:         for (int i = 0, j = 0; j < len; i++) {
136:             int f = toDigit(data[j], j) << 4;
137:             j++;
138:             f = f | toDigit(data[j], j);
139:             j++;
140:             out[i] = (byte) (f & 0xFF);
141:         }
142: 
143:         return out;
144:     }
145: 
146:     /**
147:      * Converts a hexadecimal character to an integer.
148:      *
149:      * @param ch    A character to convert to an integer digit
150:      * @param index The index of the character in the source
151:      * @return An integer
152:      * @throws IllegalArgumentException if ch is an illegal hex character
153:      */
154:     protected static int toDigit(char ch, int index) throws IllegalArgumentException {
155:         int digit = Character.digit(ch, 16);
156:         if (digit == -1) {
157:             throw new IllegalArgumentException("Illegal hexadecimal charcter " + ch + " at index " + index);
158:         }
159:         return digit;
160:     }
