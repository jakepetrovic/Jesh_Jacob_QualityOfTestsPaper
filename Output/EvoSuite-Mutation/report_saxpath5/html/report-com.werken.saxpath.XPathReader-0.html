124:     void pathExpr() throws SAXPathException
125:     {
126:         getXPathHandler().startPathExpr();
127: 
128:         switch ( LA(1) )
129:         {
130:             case INTEGER:
131:             case DOUBLE:
132:             case LITERAL:
133:             case LEFT_PAREN:
134:             case DOLLAR:
135:             {
136:                 filterExpr();
137: 
138:                 if ( LA(1) == SLASH || LA(1) == DOUBLE_SLASH )
139:                 {
140:                     locationPath( false );
141:                 }
142: 
143:                 break;
144:             }
145:             case IDENTIFIER:
146:             {
147:                 // if ( ( ( LA(2) == LEFT_PAREN )
148:                   //      &&
149:                   //      ( ! isNodeTypeName( LT(1) ) ) ) )

151:                 if ( ( LA(2) == LEFT_PAREN
152:                      &&
153:                        ! isNodeTypeName( LT(1) ) )
154:                      ||
155:                     ( LA(2) == COLON
156:                       &&
157:                       LA(4) == LEFT_PAREN) ) 
158:                 {
159:                     filterExpr();
160:                     
161:                     if ( LA(1) == SLASH || LA(1) == DOUBLE_SLASH)
162:                     {
163:                         locationPath( false );
164:                     }
165: 
166:                     break;
167:                 }
168:                 else
169:                 {
170:                     locationPath( false );
171:                     break;
172:                 }
173:             }
174:             case DOT:
175:             case DOT_DOT:
176:             case STAR:
177:             case AT:
178:             {
179:                 locationPath( false );
180:                 break;
181:             }
182:             case SLASH:
183:             case DOUBLE_SLASH:
184:             {
