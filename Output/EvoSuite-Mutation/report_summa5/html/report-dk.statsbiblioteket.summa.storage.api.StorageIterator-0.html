 75:     public StorageIterator(ReadableStorage iteratorHolder, long key, int maxBufferSize) {

 77:         this.iteratorHolder = iteratorHolder;
 78:         this.key = key;

 84:         }
 85:     }

 87:     @Override
 88:     public boolean hasNext() {
 89:         try {
 90:             checkRecords();
 91:         } catch (IOException e) {
 92:             log.warn ("Failed to retrieve records: " + e.getMessage(), e);
 93:             next = false;
 94:         }
 95:

 97:     }

116:     /**
117:      * Download next batch of records if applicable and set the 'next' state appropriately
118:      *
119:      * @throws IOException if something went wrong when checking records.

121:     private void checkRecords () throws IOException {
122:         if (records.isEmpty() && next) {
123:             try {

125:                 List<Record> recs = iteratorHolder.next(key, maxQueueSize);

129:                 if (recs.size() < maxQueueSize) {
