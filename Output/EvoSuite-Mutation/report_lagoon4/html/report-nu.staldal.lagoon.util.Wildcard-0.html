 45: /**
 46:  * Wildcard processing methods. All methods in this class are static.
 47:  *
 48:  * A wildcard pattern consists of one or more simple patterns, separated by ';'.
 49:  * A simple pattern may contain one '*' which match any string 
 50:  * (including the empty string).
 51:  */
 52: public final class Wildcard
 53: {
 54: 	/**
 55: 	 * Private default constructor to prevent instantiation.
 56: 	 */
 57: 	private Wildcard() {}
 58: 
 59:     /**
 60:      * Check if a wildcard pattern can possibly match more than one filename.
 61:      *
 62:      * @param pattern  the wildcard pattern to test
 63:      */
 64:     public static boolean isWildcard(String pattern)
 65:     {
 66:         return (pattern.indexOf('*') >= 0) || (pattern.indexOf(';') >= 0);
 67:     }
 68: 
 69:     /**
 70:      * Attempt to match a filename matches to wildcard pattern.
 71:      *
 72:      * @param pattern   the wildcard pattern
 73:      * @param filename  the filename
 74:      *
 75:      * @return the string substituted into the pattern,
 76:      *         or null if no match could be made
 77:      */
 78:     public static String matchWildcard(String pattern, String filename)
 79:     {
 80:         StringTokenizer st = new StringTokenizer(pattern, ";");
 81: 
 82:         while (st.hasMoreTokens())
 83:         {
 84:             String pat = st.nextToken();
 85:             int star = pat.indexOf('*');
 86: 
 87:             if (star < 0)
 88:             {
 89:                 if (pat.equals(filename)) return "";
 90:             }
 91:             else
 92:             {
 93:                 if (!filename.startsWith(pat.substring(0, star)))
 94:                     continue;
 95: 
 96:                 if (!filename.endsWith(pat.substring(star+1)))
 97:                     continue;
 98: 
 99:                 return filename.substring(star,
100:                     star+filename.length()-pat.length()+1);
101:             }
102:         }
103: 
104:         return null;
105:     }
106: 
107:     /**
108:      * Instantiate a wildcard pattern to a filename.
109:      *
110:      * @param pattern   the wildcard pattern
111:      * @param part      the string to substitue into the pattern
112:      *                  (returned from matchWildcard)
113:      *
114:      * @return  the filename
115:      *
116:      * @see #matchWildcard
117:      */
118:     public static String instantiateWildcard(String pattern, String part)
119:     {
120:         int star = pattern.indexOf('*');
121: 
122:         if (star < 0)
123:             return pattern;
124: 
125:         return pattern.substring(0, star) + part + pattern.substring(star+1);
126:     }
127: }
