 48:      * already. The absolute location will be relative to the System property
 49:      * "summa.control.client.persistent.dir". If that system property does not
 50:      * exist, the location will be relative to the current dir.
 51:      *
 52:      * @param file a file that should be relative to the persistent dir.
 53:      * @return the file relative to the persistent dir.
 54:      */
 55:     public static File getPersistentFile(File file) {
 56:         if (file == null) {
 57:             log.warn("Got null in getPersistentFile. Returning null");
 58:             return null;
 59:         }
 60:         if (file.getPath().equals(file.getAbsolutePath())) {
 61:             log.trace("getPersistentFile(" + file.getPath() + ") got a file that was already absolute");
 62:             return file;
 63:         }
 64:         String persistentBase;
 65:         try {
 66:             persistentBase = System.getProperty(SYSPROP_PERSISTENT_DIR);
 67:         } catch (NullPointerException e) {
 68:             //noinspection DuplicateStringLiteralInspection
 69:             log.warn("System property '" + SYSPROP_PERSISTENT_DIR + "' not defined");
 70:             return file;
 71:         } catch (SecurityException e) {
 72:             //noinspection DuplicateStringLiteralInspection
 73:             log.warn("The SecurityManager disallows the extraction of system property '" + SYSPROP_PERSISTENT_DIR
 74:                      + "'");
 75:             return file;
 76:         }
 77:         log.trace("Resolved system property '" + SYSPROP_PERSISTENT_DIR + "' to '" + persistentBase + "'");
 78:         if (persistentBase == null || "".equals(persistentBase)) {
 79:             log.debug("No persistentBase specified for system property " + SYSPROP_PERSISTENT_DIR
 80:                       + ". Returning path '" + file.getAbsolutePath() + "' directly");
 81:             return file.getAbsoluteFile();
 82:         }
 83:         try {
 84:             File persistentBaseFile = new File(persistentBase);
 85:             File resolved = new File(persistentBaseFile, file.getPath()).getAbsoluteFile();

 87:             return resolved;
