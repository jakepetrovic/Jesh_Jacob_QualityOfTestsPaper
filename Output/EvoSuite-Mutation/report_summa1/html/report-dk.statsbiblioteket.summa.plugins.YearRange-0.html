 21: /**
 22:  * This plugin is used to generate ranges of years.
 23:  */
 24: @QAInfo(level = QAInfo.Level.NORMAL,
 25:         state = QAInfo.State.IN_DEVELOPMENT,
 26:         author = "hal")
 27: public class YearRange {
 28: 
 29:     private static final Log log = LogFactory.getLog(YearRange.class);
 30: 
 31:     /**
 32:      * A thread local StringBuffer that is reset on each get() request
 33:      */
 34:     private static final ThreadLocal<StringBuffer> threadLocalBuffer = new ThreadLocal<StringBuffer>() {
 35:         @Override
 36:         protected StringBuffer initialValue() {
 37:             return new StringBuffer();
 38:         }
 39: 
 40:         @Override
 41:         public StringBuffer get() {
 42:             StringBuffer buf = super.get();
 43:             buf.setLength(0); // clear/reset the buffer
 44:             return buf;
 45:         }
 46: 
 47:     };
 48: 
 49: 
 50:     /**
 51:      * Convert a string to a range of years if possible.<br>
 52:      * The input needs to have <code>in.lenght() == 4</code> and may contain one
 53:      * or two tailing <code>?</code><br>
 54:      * if the input does not meet theese conditions the input is returned
 55:      * unchanged.
 56:      *
 57:      * @param in the string to expand
 58:      * @return a expanded string representing a year range ex:
 59:      *         199? -> 1990 1991 1992 1993 1994 1995 1996 1997 1998 1999
 60:      */
 61:     public static String makeRange(String in) {
 62:         try {
 63:             return doit(in, in);
 64:         } catch (Exception e) {
 65:             log.warn(e.getMessage());
 66:             return in;
 67:         }
 68:     }
 69: 
 70:     /**
 71:      * Make a range expansion between the two inputs. The <code>?<code>
 72:      * character can be used as a tailing wildcard (up to two) for both inputs.
 73:      * <p/>
 74:      * The expanded range will expand between the lowest and highest number
 75:      * regardless of the input order.
 76:      *
 77:      * @param in1 a bounderey for the range.
 78:      * @param in2 a bounderey for the range.
 79:      * @return an expanded range.
 80:      */
 81:     public static String makeRange(String in1, String in2) {
 82:         try {
 83:             return doit(in1, in2);
 84:         } catch (Exception e) {
 85:             log.debug("When making YearRange for ('" + in1 + "', '" + in2 + "'): " + e.getMessage());
 86:             return in1 + "-" + in2;
 87:         }
 88:     }
 89: 
 90:     //expand input to range of ints concatinated in String
 91:     private static String doit(String in1, String in2) {
 92: 
 93:         if (in1.length() > 4 || in2.length() > 4) {
 94:             throw new IllegalArgumentException("Input length greater than 4");
 95:         }
 96: 
 97:         StringBuffer buf = threadLocalBuffer.get();
 98: 
 99:         if (in1.equals(in2) && (in1.contains("???") || in1.contains("????"))) {
100:             return in1;
101:         }
102: 
103:         String years[] = new String[4];
104:         years[0] = in1;
105:         years[1] = in1;
106:         years[2] = in2;
107:         years[3] = in2;
108:         if (in1.contains("?")) {
109:             years[0] = in1.replace("?", "0");
110:             years[1] = in1.replace("?", "9");
111:         }
