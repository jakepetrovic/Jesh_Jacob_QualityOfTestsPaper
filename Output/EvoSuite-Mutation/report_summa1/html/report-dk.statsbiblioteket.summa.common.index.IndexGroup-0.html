 48:      * @param name the name of the group.

 70:      * True if the given groupName matches the name of the group or

 72:      *
 73:      * @param groupName the name to match.

 75:      * @return true if the name matches this group.
 76:      */
 77:     public boolean isMatch(String groupName, String lang) {
 78:         if (name.equals(groupName)) {
 79:             return true;
 80:         }
 81:         for (IndexAlias alias : aliases) {
 82:             if (alias.isMatch(groupName, lang)) {
 83:                 return true;
 84:             }
 85:         }
 86:         return false;
 87:     }

 98:         //noinspection DuplicateStringLiteralInspection
 99:         log.trace("Adding Alias '" + alias + "' to group '" + name + "'");

104:      * @return the aliases for this group.
105:      */
106:     public Set<IndexAlias> getAliases() {

108:     }

112:      * The order of priority is direct match first, the alias-match.

114:      * @param name the name to search for.
115:      * @param lang the language to use for alias-match. If lang == null,

178:     /**
179:      * Create a group based on the given Document Node. The Node should conform
180:      * to the output from {@link #toXMLFragment()}.
181:      * </p><p>
182:      *
183:      * @param node          a representation of a Field.
184:      * @param fieldProvider if any fields are specified, the fieldProvider is
185:      *                      queried for the parent.
186:      * @return the name of the newly group.
187:      * @throws ParseException if the node could not be parsed.
188:      *                        The index of the exception will normally be -1.
189:      */
190:     private String parse(Node node, FieldProvider<F> fieldProvider) throws ParseException {
191:         //noinspection DuplicateStringLiteralInspection
192:         log.trace("parse called");
193:         Node groupNameNode = node.getAttributes().getNamedItem("name");
194:         if (groupNameNode == null) {
195:             throw new ParseException("No name specified for group", -1);
196:         }
197:         String name = groupNameNode.getNodeValue();
198:         log.trace("parse: Located group name '" + name + "'");
199:         aliases = IndexAlias.getAliases(node);
200: 
201:         NodeList children = node.getChildNodes();
202:         for (int i = 0; i < children.getLength(); i++) {
203:             Node child = children.item(i);
204:             //log.debug("Found child '" + child.getNodeName() + "', in group");
205:             //noinspection DuplicateStringLiteralInspection
206:             if (child.getNodeName() != null && "field".equals(child.getNodeName())) {
207:                 Node fieldNameNode = child.getAttributes().getNamedItem("ref");
208:                 if (fieldNameNode == null
209:                     || "".equals(fieldNameNode.getNodeValue())) {
210:                     //noinspection DuplicateStringLiteralInspection
211:                     log.warn(String.format(
212:                             "Undefined field name in group '%s'. Skipping",
213:                             name));
214:                     continue;
215:                 }
216:                 String fieldRef = fieldNameNode.getNodeValue();
217:                 log.trace("Found field ref '" + fieldRef + " in group '"
218:                           + name + "'");
219:                 F field = fieldProvider.getField(fieldRef);
220:                 if (field == null) {
221:                     throw new ParseException(String.format(
222:                             "The field '%s' in group '%s' did not exist",
223:                             fieldRef, name), -1);
224:                 }
225:                 addField(field);
226:             }
227:         }
228:         log.debug("Resolved " + fields.size() + " fields for group " + name + ": " + Strings.join(fields, ", "));
229:         return name;
230:     }
