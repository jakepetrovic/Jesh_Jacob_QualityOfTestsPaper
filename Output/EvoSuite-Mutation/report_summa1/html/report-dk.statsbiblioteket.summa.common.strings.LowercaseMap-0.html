 25: /**
 26:  * Has a base map from char -> char which converts all chars from 0 to 65535 to
 27:  * lowercase. Given a map char -> char or char -> char[], and ensures that all
 28:  * rules handles both upper- and lowercase to the same result. Also ensures that
 29:  * the result is always lowercase.
 30:  */
 31: @QAInfo(level = QAInfo.Level.NORMAL,
 32:         state = QAInfo.State.IN_DEVELOPMENT,
 33:         author = "te")
 34: public class LowercaseMap {
 35:     private static Log log = LogFactory.getLog(LowercaseMap.class);
 36: 
 37:     private static Map<String, String> LOWER;
 38:     static { // Create lower
 39:         LOWER = new HashMap<String, String>(65535);
 40:         for (int cInt = 0 ; cInt < Character.MAX_VALUE ; cInt++) {
 41:             String s = String.valueOf((char)cInt);
 42:             if (!toLower(s).equals(s)) {
 43:                 LOWER.put(s, toLower(s));
 44:             }
 45:         }
 46:         log.trace("Finished constructing base TOLOWER with " + LOWER.size() + " rules");
 47:     }
 48: 
 49:     // Technically speaking we should do this with a custom Locale.
 50:     private static String toLower(String in) {
 51:         return in.toLowerCase(Locale.ENGLISH);
 52:     }
 53: 
 54:     /**
 55:      * Extends and updates the rules in the given replaceMap so that it converts
 56:      * every output to lowercase. For all chars without rules, a new rule is
 57:      * added if the value should be different from the source.
 58:      * </p><p>
 59:      * The replaceMap itself is not changed.
 60:      * @param replaceMap a map with replacements. The keys must be of length 1.
 61:      * @return a map conforming to the given rules.
 62:      * @throws IllegalArgumentException if one or more keys are not of length 1.
 63:      */
 64:     public static Map<String, String> getLowercaseMap(
 65:             Map<String, String> replaceMap) {
 66:         Map<String, String> result = new HashMap<String, String>(replaceMap.size() * 2 + LOWER.size());
 67:         // Handle existing
 68:         for (Map.Entry<String, String> entry: replaceMap.entrySet()) {
 69:             String key = entry.getKey();
 70:             String value = toLower(entry.getValue());
 71:             if (!key.equals(toLower(key))) {
 72:                 // Check for alternative value
 73:                 String lowerValue = toLower(replaceMap.get(toLower(key)));
 74: //                if (!lowerValue.)
 75:             }
 76:             // Ensure destination is lowercase
 77:             if (!value.equals(toLower(value))) {
 78:                 value = toLower(value);
 79:                 entry.setValue(toLower(entry.getValue()));
 80:             }
 81:             // Check for both upper- and lowercase keys
 82:         }
 83: 
 84:         // Fill missing
 85:         for (Map.Entry<String, String> entry: LOWER.entrySet()) {
 86:             if (!replaceMap.containsKey(entry.getKey())) {
 87:                 replaceMap.put(entry.getKey(), entry.getValue());
 88:             }
 89:         }
 90:         return null;
 91:     }
