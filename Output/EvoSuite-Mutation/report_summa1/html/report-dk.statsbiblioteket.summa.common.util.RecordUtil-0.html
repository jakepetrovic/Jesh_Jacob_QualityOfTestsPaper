209:     private static void toXML(XMLStreamWriter out, int level, Set<Record> processed, Record record,
210:                               boolean escapeContent, String timing) throws XMLStreamException {
211:         if (processed.contains(record)) {
212:             return;
213:         }
214:         processed.add(record);
215:         log.trace("Constructing inner XML for Record '" + record.getId() + "'");
216:         String indent = "";
217:         while (indent.length() < level * 2) {
218:             indent += "  ";
219:         }
220:         if (level != 0) {
221:             out.writeCharacters("\n");
222:         }
223:         out.writeStartElement(RECORD);
224:         if (level == 0) {
225:             out.writeNamespace("", RECORD_NAMESPACE);
226:             if (timing != null && !"".equals(timing)) {
227:                 out.writeAttribute("timing", timing);
228:             }
229:         }
230:         out.writeAttribute(ID, record.getId());
231:         out.writeAttribute(BASE, record.getBase());
232:         out.writeAttribute(DELETED, Boolean.toString(record.isDeleted()));
233:         out.writeAttribute(INDEXABLE, Boolean.toString(record.isIndexable()));
234:         synchronized (schemaTimestampFormatter) {
235:             out.writeAttribute(CTIME, schemaTimestampFormatter.format(new Date(record.getCreationTime())));
236:             out.writeAttribute(MTIME, schemaTimestampFormatter.format(new Date(record.getModificationTime())));
237:         }
238:         out.writeCharacters("\n");
239: 
240:         out.writeStartElement(CONTENT);
241:         try {
242:             writeContent(out, record, escapeContent);
243:         } catch (RuntimeException e) {
244:             throw new XMLStreamException(String.format("Unable to write XML for content from %s", record), e);
245:         }
246:         out.writeEndElement();
247: 
248:         if (record.getParents() != null && !record.getParents().isEmpty()) {
249:             out.writeCharacters("\n");
250:             out.writeStartElement(PARENTS);
251:             for (Record parent: record.getParents()) {
252:                 toXML(out, level+1, processed, parent, escapeContent);
253:             }
254:             out.writeCharacters("\n");
255:             out.writeEndElement();
256:         }
257: 
258:         if (record.getChildren() != null && !record.getChildren().isEmpty()) {
259:             out.writeCharacters("\n");
260:             out.writeStartElement(CHILDREN);
261:             for (Record child: record.getChildren()) {
262:                 if (log.isTraceEnabled()) {
263:                     log.trace("Calling toXML on " + child);
264:                 }
265:                 toXML(out, level+1, processed, child, escapeContent);

272:             out.writeCharacters("\n");
273:             out.writeStartElement(META);

285:         out.writeCharacters("\n");

390:     private static Record processRecord(XMLStreamReader reader) throws ParseException, XMLStreamException {
391:         int eventType = reader.getEventType();
392:         if (!(eventType == XMLEvent.START_ELEMENT
393:             && RECORD.equals(reader.getLocalName())
394:             && RECORD_NAMESPACE.equals(reader.getName().getNamespaceURI()))) {
395:             throw new ParseException(String.format(
396:                     "The element should be %s:%s, but was %s:%s",
397:                     RECORD_NAMESPACE, RECORD,
398:                     reader.getName().getNamespaceURI(), reader.getLocalName()),
399:                     reader.getLocation().getCharacterOffset());
400:         }
401: 
402:         Record record;
403:         long mtime;
404:         try {
405:             //noinspection DuplicateStringLiteralInspection
406:             String id = getAttributeValue(reader, RECORD_NAMESPACE, ID);
407:             String base = getAttributeValue(reader, RECORD_NAMESPACE, BASE);
408:             boolean deleted = Boolean.parseBoolean(getAttributeValue(reader, RECORD_NAMESPACE, DELETED));
409:             boolean indexable = Boolean.parseBoolean(getAttributeValue(reader, RECORD_NAMESPACE, INDEXABLE));
410:             long ctime = getDatetimeValue(reader, RECORD_NAMESPACE, CTIME);
411:             mtime = getDatetimeValue(reader, RECORD_NAMESPACE, MTIME);
412:             log.trace("Extracted record attributes for Record '" + id + "'");
413:             record = new Record(id, base, DUMMY_CONTENT);
414:             record.setDeleted(deleted);
415:             record.setIndexable(indexable);
416:             record.setCreationTime(ctime);
417:         } catch (Exception e) {
418:             throw new XMLStreamException("Exception extracting attributes", e);
419:         }
420: 
421:         log.trace("Looking for content, parents, children and meta for Record '" + record.getId() + "'");
422:         while (reader.hasNext()) {
423:             reader.next();
424:             if (reader.getEventType() == XMLStreamReader.END_ELEMENT
425:                 && RECORD.equals(reader.getLocalName())
426:                 && RECORD_NAMESPACE.equals(reader.getName().getNamespaceURI())) {
427:                 break;
428:             }
429:             if (reader.getEventType() != XMLStreamReader.START_ELEMENT) {
430:                 if (reader.getEventType() == XMLStreamReader.CHARACTERS
431:                     || reader.getEventType() == XMLStreamReader.COMMENT
432:                     || reader.getEventType() == XMLStreamReader.SPACE) {
433:                     // Ignore text and comments in between fields
434:                 } else {
435:                     log.debug("Expected start-elementor chars but got " + XMLUtil.eventID2String(
436:                             reader.getEventType()) + " in " + record.getId());
437:                 }
438:                 continue;
439:             }
440: 
441:             if (reader.getEventType() != XMLStreamReader.START_ELEMENT) {
442:                 log.debug(String.format(
443:                         "processRecord: Expected START_ELEMENT but got %s",
444:                         XMLUtil.eventID2String(reader.getEventType())));
445:                 continue;
446:             }
447: 
448:             if (CONTENT.equals(reader.getLocalName()) &&
449:                 RECORD_NAMESPACE.equals(reader.getName().getNamespaceURI())) {
450:                 log.trace("Found content for Record '" + record.getId() + "'");
451:                 try {
452:                     record.setContent(getContent(reader, record), false);
453:                 } catch (UnsupportedEncodingException e) {
454:                     //noinspection DuplicateStringLiteralInspection
455:                     throw new IllegalStateException("utf-8 not supported", e);
456:                 }
457:                 continue;
458:             }
459: 
460:             if (PARENTS.equals(reader.getLocalName()) &&
461:                 RECORD_NAMESPACE.equals(reader.getName().getNamespaceURI())) {
462:                 log.trace("Found parent element for Record '" + record.getId() + "'");
463:                 record.setParents(getRelatives(reader, PARENTS));
464:                 continue;
465:             }
466: 
467:             if (CHILDREN.equals(reader.getLocalName()) &&
468:                 RECORD_NAMESPACE.equals(reader.getName().getNamespaceURI())) {
469:                 log.trace("Found children element for Record '" + record.getId() + "'");
470:                 record.setChildren(getRelatives(reader, CHILDREN));
471:                 continue;
472:             }
473: 
474:             if (META.equals(reader.getLocalName()) &&
475:                 RECORD_NAMESPACE.equals(reader.getName().getNamespaceURI())) {
476:                 log.trace("Found meta element for Record '" + record.getId() + "'");
477:                 record.setMeta(getMeta(reader));
478:                 continue;
479:             }

481:             log.debug("Encountered unexpected start element "
482:                       + reader.getName().getNamespaceURI() + ":"
483:                       + reader.getLocalName());
484:         }

486:         record.setModificationTime(mtime);
487:         return record;
488:     }

490:     private static byte[] getContent(XMLStreamReader reader, Record record)
491:                        throws XMLStreamException, UnsupportedEncodingException {
492:         log.info(XMLUtil.eventID2String(reader.getEventType()));
493:         for (int i = 0 ; i < reader.getAttributeCount() ; i++) {
494:             if (reader.getAttributeLocalName(i).equals(CONTENT_TYPE)) {
495:                 String type = reader.getAttributeValue(i);
496:                 if (type.equals(CONTENT_TYPE_XML)) {
497:                     StringWriter sw = new StringWriter(5000);
498:                     XMLStreamWriter writer = xmlOutputFactory.createXMLStreamWriter(sw);
499:                     reader.next();
500:                     writer.writeStartDocument("UTF-8", "1.0");
501:                     copyContent(reader, writer, record, true, -1);
502:                     writer.writeEndDocument();
503:                     return sw.toString().getBytes("utf-8");

505:                     log.warn(String.format(
506:                             "Encountered unknown content type '%s' for %s. Parsing as string", type, record));
507:                 }
508:             }

538:     /* Like above but with nice debug message on error is record != null */
539:     private static void copyContent(XMLStreamReader reader, XMLStreamWriter writer, Record record,
540:                                     boolean skipDeclarations, int maxElements) throws XMLStreamException {
541:         int depth = 0;
542:         int elementcount = 0;
543:         while (reader.getEventType() != XMLStreamReader.END_DOCUMENT) {
544:             switch (reader.getEventType()) {
545:                 case XMLStreamReader.END_ELEMENT: {
546:                     depth--;
547:                     if (depth < 0) {
548:                         return; // Too far up
549:                     }
550:                     writer.writeEndElement();
551:                     break;
552:                 }
553:                 case XMLStreamReader.CHARACTERS:
554:                 case XMLStreamReader.SPACE: {
555:                     writer.writeCharacters(reader.getText());
556:                     break;
557:                 }
558:                 case XMLStreamReader.CDATA: {
559:                     writer.writeCData(reader.getText());
560:                     break;
561:                 }
562:                 case XMLStreamReader.COMMENT: {
563:                     writer.writeComment(reader.getText());
564:                     break;
565:                 }
566:                 case XMLStreamReader.PROCESSING_INSTRUCTION: {
567:                     writer.writeProcessingInstruction(reader.getPITarget(), reader.getPIData());
568:                     break;
569:                 }
570:                 case XMLStreamReader.ENTITY_REFERENCE: {
571:                     // TODO: Verify that this part is correct
572:                     writer.writeEntityRef(reader.getLocalName());
573:                     break;
574:                 }
575:                 case XMLStreamReader.DTD: {
576:                     if (!skipDeclarations) {
577:                         writer.writeDTD(reader.getText());
578:                     }
579:                     break;
580:                 }
581:                 case XMLStreamReader.START_DOCUMENT: {
582:                     if (!skipDeclarations) {
583:                         // TODO: Don't we need to transfer more information?
584:                         writer.writeStartDocument(reader.getVersion());
585:                     }
586:                     break;
587:                 }
588:                 case XMLStreamReader.NAMESPACE: {
589:                     for (int i = 0 ; i < reader.getNamespaceCount() ; i++) {
590:                         writer.writeNamespace(reader.getNamespacePrefix(i), reader.getNamespaceURI(i));
591:                     }
592:                     break;
593:                 }
594:                 case XMLStreamReader.START_ELEMENT: {
595:                     if (elementcount == maxElements) {
596:                         return;
597:                     }
598:                     elementcount++;
599:                     String namespaceURI = reader.getNamespaceURI();
600:                     if (namespaceURI != null) {
601:                         writer.writeStartElement( reader.getPrefix(), reader.getLocalName(), reader.getNamespaceURI());
602:                     } else {
603:                         writer.writeStartElement(reader.getLocalName());
604:                     }
605:                     // TODO: What about default namespace?
606:                     for (int i = 0 ; i < reader.getNamespaceCount() ; i++) {
607:                         writer.writeNamespace(reader.getNamespacePrefix(i), reader.getNamespaceURI(i));
608:                     }
609:                     for (int i = 0 ; i < reader.getAttributeCount() ; i++) {
610:                         if (reader.getAttributeNamespace(i) == null) {
611:                             writer.writeAttribute(reader.getAttributeLocalName(i), reader.getAttributeValue(i));
612:                             continue;
613:                         }
614:                         writer.writeAttribute(reader.getAttributePrefix(i), reader.getAttributeNamespace(i),
615:                                               reader.getAttributeLocalName(i), reader.getAttributeValue(i));
616:                     }
617:                     depth++;
618:                     break;
619:                 }
620:                 default: {
621:                     throw new XMLStreamException(String.format(
622:                             "Unknown event type %d from reader", reader.getEventType()));
623:                 }
624:             }
625:             try {
626:                 reader.next();
627:             } catch (XMLStreamException e) {
628:                 if (record == null) {
629:                     throw new XMLStreamException("Parse error (content could not be extracted for debugging)",
630:                                                  e.getLocation(), e);
631:                 }
632:                 String content = record.getContent() == null ? "[no content]" : record.getContentAsUTF8();
633:                 throw new XMLStreamException(String.format(
634:                         "Parse error. First 20 characters of %s was '%s'. Error was '%s'",
635:                         record, content.substring(0, Math.min(20, content.length())),
636:                         e.getMessage()), e.getLocation(), e);
637:             }
638:         }
639:     }

707:      * Calculates the approximate memory usage of the Record.

709:      * This method is fairly light-weight and aims for approximations in the
710:      * kilobyte range. Thus the length of String attributes et al are ignored.
711:      * The sizes is primarily determined by content and meta-data.
712:      * @param record      the Record to calculate memory usage for.

714:      *                    children is performed, summing the sizes.

721:     private static int calculateRecordSize(Record record, boolean followLinks, Set<Record> visited) {
722:         if (visited != null) {
723:             if (visited.contains(record)) {
724:                 return 0;
725:             }
726:             visited.add(record);
727:         }
728: 
729:         int size = record.getContent(false) == null ? 0
730:                    : record.getContent(false).length;
731:         if (record.getMeta() != null) {
732:             for (Map.Entry<String, String> entry: record.getMeta().entrySet()) {
733:                 try {
734:                     size += entry.getKey().length() + entry.getValue().length();
735:                 } catch (NullPointerException e) {
736:                     log.trace("NPE while requesting key and value size from entry " + entry.getKey() + " in " + record,
737:                               e);
738:                 }
739:             }
740:         }
741:         if (!followLinks || record.getParents() == null && record.getChildren() == null) {
742:             return size;
743:         }
744:         if (visited == null) {
745:             visited = new HashSet<Record>(10);
746:             visited.add(record);
747:         }
748:         if (record.getParents() != null) {
749:             for (Record parent: record.getParents()) {
750:                 size += calculateRecordSize(parent, followLinks, visited);
751:             }
752:         }
753:         if (record.getChildren() != null) {
754:             for (Record child: record.getChildren()) {
755:                 size += calculateRecordSize(child, followLinks, visited);
756:             }
757:         }
758:         return size;
759:     }
