307:     private MiniConnectionPoolManager.StatementHandle generateStatementHandle(String sql) {
308:         MiniConnectionPoolManager.StatementHandle sh = statements.get(sql);
309:         if (sh != null) {
310:             return sh;
311:         }
312:         try {
313:             sh = prepareStatement(sql);
314:         } catch (SQLException e) {
315:             throw new IllegalArgumentException(
316:                 "The received SQL could not be prepared. SQL statement was '" + sql + "'", e);
317:         }
318:         statements.put(sql, sh);
319:         log.debug("Generated new StatementHandle for sql '" + sql + "'");
320:         return sh;
321:     }

323:     private String NULL_ATTRIBUTES = null;
324:     private String NULL_ATTRIBUTES_RELATIONS = null;
325:     /**
326:      * Produces a SQL-query part meant for requesting the fields matching the given attributes in options.
327:      * Example result: {@code summa_records.id,'' AS base,summa_records.deleted...}. All attributes are always
328:      * present, although those missing from options are represented with an {@code '' AS attribute}.
329:      * @param options the source of the attributes. If null, all fields are requested.
330:      * @return SQL for retrieving content from the columns corresponding to the attributes in options.
331:      */
332:     private String getColumns(QueryOptions options, boolean noRelations) {
333:         if (options == null || !options.newRecordNeeded()) {
334:             if (noRelations && NULL_ATTRIBUTES != null) {
335:                 return NULL_ATTRIBUTES;
336:             }
337:             if (!noRelations && NULL_ATTRIBUTES_RELATIONS != null) {
338:                 return NULL_ATTRIBUTES_RELATIONS;
339:             }
340:         }
341:         StringWriter records = new StringWriter(200);
342:         boolean first = true;
343:         for (QueryOptions.ATTRIBUTES attribute: QueryOptions.ATTRIBUTES.values()) {
344: /*            if (noRelations &&
345:                 (attribute == QueryOptions.ATTRIBUTES.PARENTS || attribute == QueryOptions.ATTRIBUTES.CHILDREN)) {
346:                 continue;
347:             }*/
348:             if (first) {
349:                 first = false;
350:             } else {
351:                 records.append(",");
352:             }
353: 
354:             boolean found = options == null || !options.newRecordNeeded();
355:             if (!found) {
356:                 for (QueryOptions.ATTRIBUTES queryAttribute: options.getAttributes()) {
357:                     if (attribute == queryAttribute) {
358:                         if (!(noRelations && (attribute == QueryOptions.ATTRIBUTES.PARENTS ||
359:                                               attribute == QueryOptions.ATTRIBUTES.CHILDREN))) {
360:                             found = true;
361:                             break;
362:                         }
363:                     }
364:                 }
365:             }
366:             if (noRelations && (attribute == QueryOptions.ATTRIBUTES.PARENTS ||
367:                                 attribute == QueryOptions.ATTRIBUTES.CHILDREN)) {
368:                 found = false;
369:             }
370:             if (found) {
371:                 if (attribute == QueryOptions.ATTRIBUTES.PARENTS || attribute == QueryOptions.ATTRIBUTES.CHILDREN) {
372:                     records.append(DatabaseStorage.RELATIONS);
373:                 } else {
374:                     records.append(DatabaseStorage.RECORDS);
375:                 }
376:                 records.append(".");
377:             } else {
378:                 records.append("'' AS ");
379:             }
380:             records.append(attributeToField(attribute));
381:         }
382:         if (options == null || !options.newRecordNeeded()) {
383:             if (noRelations && NULL_ATTRIBUTES != null) {
384:                 NULL_ATTRIBUTES = records.toString();
385:             }
386:             if (!noRelations && NULL_ATTRIBUTES_RELATIONS != null) {
387:                 NULL_ATTRIBUTES_RELATIONS = records.toString();
388:             }
389:         }
390:         if (log.isTraceEnabled()) {
391:             log.trace("Generated columns '" + records.toString());
392:         }
393:         return records.toString();
394:     }
