 33: /**
 34:  * Object builder that uses reflection and Apache Commons BeanUtils to build objects given a
 35:  * map of "property values".  Typically these come from the JSecurity INI configuration and are used
 36:  * to construct or modify the SecurityManager, its dependencies, and web-based security filters.
 37:  *
 38:  * @author Les Hazlewood
 39:  * @author Jeremy Haile
 40:  * @since 0.9
 41:  */

 49:     private static final String OBJECT_REFERENCE_BEGIN_TOKEN = "$";
 50:     private static final String ESCAPED_OBJECT_REFERENCE_BEGIN_TOKEN = "\\$";
 51:     private static final String GLOBAL_PROPERTY_PREFIX = "jsecurity";

 56:     public ReflectionBuilder() {
 57:         setObjects(new LinkedHashMap<String, Object>());
 58:     }

 60:     public ReflectionBuilder(Map defaults) {
 61:         setObjects(defaults);
 62:     }

 65:         return objects;
 66:     }

 72:     public Map buildObjects(Map<String, String> kvPairs) {
 73:         if (kvPairs != null && !kvPairs.isEmpty()) {
 74: 
 75:             // Separate key value pairs into object declarations and property assignment
 76:             // so that all objects can be created up front
 77:             Map<String, String> instanceMap = new HashMap<String, String>();
 78:             Map<String, String> propertyMap = new HashMap<String, String>();
 79:             for (Map.Entry<String, String> entry : kvPairs.entrySet()) {
 80:                 if (entry.getKey().indexOf('.') < 0 || entry.getKey().endsWith(".class")) {
 81:                     instanceMap.put(entry.getKey(), entry.getValue());
 82:                 } else {
 83:                     propertyMap.put(entry.getKey(), entry.getValue());
 84:                 }
 85:             }
 86: 
 87:             // Create all instances
 88:             for (Map.Entry<String, String> entry : instanceMap.entrySet()) {
 89:                 createNewInstance(objects, entry.getKey(), entry.getValue());
 90:             }
 91: 
 92:             // Set all properties
 93:             for (Map.Entry<String, String> entry : propertyMap.entrySet()) {
 94:                 applyProperty(entry.getKey(), entry.getValue(), objects);
 95:             }
 96:         }
 97: 
 98:         return objects;
 99:     }
100: 
101:     protected void createNewInstance(Map objects, String name, String value) {
102: 
103:         Object currentInstance = objects.get(name);
104:         if (currentInstance != null) {
105:             log.info("An instance with name [" + name + "] already exists.  " +
106:                     "Redefining this object as a new instance of type [" + value + "].");
107:         }
108: 
109:         Object instance;//name with no property, assume right hand side of equals sign is the class name:
110:         try {
111:             instance = ClassUtils.newInstance(value);
112:             if (instance instanceof Nameable) {
113:                 ((Nameable) instance).setName(name);
114:             }
115:         } catch (Exception e) {
116:             String msg = "Unable to instantiate class [" + value + "] for object named '" + name + "'.  " +
117:                     "Please ensure you've specified the fully qualified class name correctly.";
118:             throw new ConfigurationException(msg, e);
119:         }
120:         objects.put(name, instance);
121:     }
122: 
123:     protected void applyProperty(String key, String value, Map objects) {
124: 
125:         int index = key.indexOf('.');
126: 
127:         if (index >= 0) {
128:             String name = key.substring(0, index);
129:             String property = key.substring(index + 1, key.length());
130: 
131:             if (GLOBAL_PROPERTY_PREFIX.equalsIgnoreCase(name)) {
132:                 applyGlobalProperty(objects, property, value);
133:             } else {
134:                 applySingleProperty(objects, name, property, value);
135:             }
136: 
137: 
138:         } else {
139:             throw new IllegalArgumentException("All property keys must contain a '.' character. " +
140:                     "(e.g. myBean.property = value)  These should already be separated out by buildObjects().");
141:         }
142:     }
143: 
144:     protected void applyGlobalProperty(Map objects, String property, String value) {
145:         for (Object instance : objects.values()) {
146:             try {
147:                 PropertyDescriptor pd = PropertyUtils.getPropertyDescriptor(instance, property);
148:                 if (pd != null) {
149:                     applyProperty(instance, property, value);
150:                 }
151:             } catch (Exception e) {
152:                 String msg = "Error retrieving property descriptor for instance " +
153:                         "of type [" + instance.getClass().getName() + "] " +
154:                         "while setting property [" + property + "]";
155:                 throw new ConfigurationException(msg, e);
156:             }
157:         }
158:     }
159: 
160:     protected void applySingleProperty(Map objects, String name, String property, String value) {
161:         Object instance = objects.get(name);
162:         if (property.equals("class")) {
163:             throw new IllegalArgumentException("Property keys should not contain 'class' properties since these " +
164:                     "should already be separated out by buildObjects().");
165: 
166:         } else if (instance == null) {
167:             String msg = "Configuration error.  Specified object [" + name + "] with property [" +
168:                     property + "] without first defining that object's class.  Please first " +
169:                     "specify the class property first, e.g. myObject.class = fully_qualified_class_name " +
170:                     "and then define additional properties.";
171:             throw new IllegalArgumentException(msg);
172: 
173:         } else {
174:             applyProperty(instance, property, value);
175:         }
176:     }

178:     protected boolean isReference(String value) {
179:         return value != null && value.startsWith(OBJECT_REFERENCE_BEGIN_TOKEN);
180:     }
181: 
182:     protected String getId(String referenceToken) {
183:         return referenceToken.substring(OBJECT_REFERENCE_BEGIN_TOKEN.length());
184:     }

186:     protected Object getReferencedObject(String id) {
187:         Object o = objects != null && !objects.isEmpty() ? objects.get(id) : null;
188:         if (o == null) {
189:             String msg = "The object with id [" + id + "] has not yet been defined and therefore cannot be " +
190:                     "referenced.  Please ensure objects are defined in the order in which they should be " +
191:                     "created and made avaialable for future reference.";
192:             throw new UnresolveableReferenceException(msg);
193:         }
194:         return o;
195:     }

197:     protected String unescapeIfNecessary(String value) {
198:         if (value != null && value.startsWith(ESCAPED_OBJECT_REFERENCE_BEGIN_TOKEN)) {
199:             return value.substring(ESCAPED_OBJECT_REFERENCE_BEGIN_TOKEN.length() - 1);
200:         }
201:         return value;
202:     }

204:     protected void applyProperty(Object object, String propertyName, String stringValue) {
205: 
206:         Object value;
207: 
208:         if (isReference(stringValue)) {
209:             String id = getId(stringValue);
210:             if (log.isDebugEnabled()) {
211:                 log.debug("Encountered object reference [" + stringValue + "].  Looking up object " +
212:                         "with id [" + id + "]");
213:             }
214:             value = getReferencedObject(id);
215:         } else {
216:             value = unescapeIfNecessary(stringValue);
217:         }
218: 
219:         try {
220:             if (log.isTraceEnabled()) {
221:                 log.trace("Applying property [" + propertyName + "] value [" + value + "] on object of type [" + object.getClass().getName() + "]");
222:             }
223:             BeanUtils.setProperty(object, propertyName, value);
224:         } catch (Exception e) {
225:             String msg = "Unable to set property [" + propertyName + "] with value [" + stringValue + "].  If " +
226:                     "'" + stringValue + "' is a reference to another (previously defined) object, please prefix it with " +
227:                     "'" + OBJECT_REFERENCE_BEGIN_TOKEN + "' to indicate that the referenced " +
228:                     "object should be used as the actual value.  " +
229:                     "For example, " + OBJECT_REFERENCE_BEGIN_TOKEN + stringValue;
230:             throw new ConfigurationException(msg, e);
231:         }
232:     }
