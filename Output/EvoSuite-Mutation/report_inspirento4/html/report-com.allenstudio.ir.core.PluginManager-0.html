 52:     public static final String PLUGIN_REGISTRY = "plugins.xml";
 53:     
 54:     private static final String PLUGIN_DTD =
 55:         "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" +
 56:         "<!-- DTD for plugin registry -->"                +
 57:         "<!ELEMENT plugins ( plugin* ) >"+
 58:         "<!ELEMENT plugin (#PCDATA) >" +
 59:         "<!ATTLIST plugin " +

 64: 
 65: 
 66:     private Document source;
 67:         
 68:     /**
 69:      * Private constructor for singleton use.
 70:      */
 71:     private PluginManager() {
 72:         try {
 73:             source = loadDocument(new File(
 74:                     ConfigurationManager.CONFIG_DIRECTORY + System.getProperty("file.separator") +
 75:                     PLUGIN_REGISTRY));
 76:         } catch (SAXException e) {
 77:             System.out.println("Fatal error! Cannot parse the plugin registry file.");
 78:             e.printStackTrace();
 79:             System.exit(-5);
 80:         } catch (IOException e) {
 81:             System.out.println("Fatal error! Cannot read the plugin registry file.");
 82:             e.printStackTrace();
 83:             System.exit(-6);
 84:         }
 85:     }
 86:     
 87:     private static Document loadDocument(File in)
 88:             throws SAXException, IOException {
 89:         DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
 90:         dbf.setIgnoringElementContentWhitespace(true);
 91:         dbf.setValidating(true);
 92:         dbf.setCoalescing(true);
 93:         dbf.setIgnoringComments(true);
 94:         
 95:         DocumentBuilder db;
 96:         try {
 97:             db = dbf.newDocumentBuilder();
 98:             db.setEntityResolver(new Resolver());
 99:             db.setErrorHandler(new EH());
100:             return db.parse(in);
101:         } catch (ParserConfigurationException e) {
102:             System.out.println("Fatal error! Parser configuration error occured.");
103:             e.printStackTrace();
104:             throw new Error(e);
105:         }
106:     }

109:         if (instance == null) {
110:             instance = new PluginManager();
111:         }
112:         
113:         return instance;

116:     /**
117:      * Gets all plugins' name.
118:      * The name of a plugin is the title attribute in the

120:      * @return All the names

122:     public String[] getAllPluginNames() {
123:         NodeList allList = source.getChildNodes();
124:         Element rootElement = (Element)allList.item(1);//rootElement is <plugins> tag
125:         
126:         NodeList plugins = rootElement.getChildNodes();
127:         int num = plugins.getLength();//# of plugins
128:         String[] pluginNames = new String[num];
129:         for (int i = 0; i < num; i++) {

131:         }
