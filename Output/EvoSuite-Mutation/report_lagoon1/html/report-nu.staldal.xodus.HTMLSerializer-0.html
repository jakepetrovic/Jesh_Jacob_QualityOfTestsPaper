160:     private void writeAttribute(String attQName, String attValue)
161:         throws IOException
162:     {
163:         if (attQName.equals("xmlns") || attQName.startsWith("xmlns:"))
164:         {
165:             // do not output XML namespace declarations
166:             return;
167:         }
168:         
169:         out.write(' ');
170:         out.write(attQName);
171:         
172:         if (booleanAttributes.contains(attQName.toLowerCase())
173:                 && attValue.equalsIgnoreCase(attQName))
174:         {
175:             // boolean attribute    
176:         }
177:         else
178:         {        
179:             out.write("=\"");
180:             
181:             if (disableOutputEscaping)
182:             {
183:                 out.write(attValue);
184:             }
185:             else
186:             {
187:                 out.enableEscaping();
188:                 for (int i = 0; i<attValue.length(); i++)
189:                 {
190:                     char c = attValue.charAt(i);
191:                     switch (c)
192:                     {
193:                     case '&':
194:                         if (i<attValue.length() && attValue.charAt(i+1)=='{')
195:                         {
196:                             // do not escape & immediately before { in HTML
197:                             out.write('&');    
198:                         }
199:                         else
200:                         {
201:                             out.write("&amp;");
202:                         }
203:                         break;
204:     
205:                     case '\"':
206:                         out.write("&quot;");
207:                         break;
208:                     
209:                     default:
210:                         out.write(c);
211:                     }
212:                 }
213:                 out.disableEscaping();
214:             }
215:             
216:             out.write('\"');
217:         }
218:     }

221:     public void startElement(String namespaceURI, String localName,
222: 			      String qName, Attributes atts)
223: 	    throws SAXException
224:     {
225:         fixTag();
226: 
227:         if (localName == null || localName.length() == 0)
228:         {
229:             localName = qName;
230:         }        
231:         int colon = localName.indexOf(':');
232:         if (colon > -1)
233:         {
234:             localName = localName.substring(colon+1);
235:         }
236:                 
237:         try {
238:             if (addDTD)
239:             {
240:                 if (outputConfig.doctype_system != null)
241:                 {
242:                     out.write("<!DOCTYPE ");
243:                     out.write(localName);
244:                     out.write(' ');
245:                     writeExternalId(outputConfig.doctype_public,
246:                         outputConfig.doctype_system);
247:                     out.write('>');
248:                     if (outputConfig.indent)
249:                     {                
250:                         newline();
251:                     }
252:                 }  
253:                 addDTD = false;
254:             }
255:             
256:             if (outputConfig.indent && (wasStartTag || wasEndTag) && inFormattedElement == 0)
257:             {
258:                 if (elementDepth > 0)
259:                 {
260:                     newline();
261:                     for (int i = 0; i<elementDepth*2; i++)
262:                     {
263:                         out.write(' ');
264:                     }
265:                 }
266:             }
267:             out.write('<');
268:             out.write(localName);
269:

273:                 if (attQName == null || attQName.length() == 0)
274:                 {

276:                 }
277:                 writeAttribute(attQName, atts.getValue(j));                
278:             }            
279:         }
280:         catch (IOException e)
281:         {
282:             throw new SAXException(e);    
283:         }            
284:

286:             inFormattedElement++;
287:         
288:         if (notEscapeElements.contains(localName.toLowerCase()))
289:             inNotEscapeElement++;
290:

293:         wasStartTag = true;
