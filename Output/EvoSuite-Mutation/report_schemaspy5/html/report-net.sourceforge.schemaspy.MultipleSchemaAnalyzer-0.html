181:     private List<String> getPopulatedSchemas(DatabaseMetaData meta, String schemaSpec, boolean isCatalog) throws SQLException {
182:         List<String> populatedSchemas;
183: 
184:         if ((!isCatalog && meta.supportsSchemasInTableDefinitions()) ||

186:             Pattern schemaRegex = Pattern.compile(schemaSpec);

188:             populatedSchemas = DbAnalyzer.getPopulatedSchemas(meta, schemaSpec, isCatalog);
189:             Iterator<String> iter = populatedSchemas.iterator();
190:             while (iter.hasNext()) {
191:                 String schema = iter.next();
192:                 if (!schemaRegex.matcher(schema).matches()) {
193:                     if (fineEnabled) {
194:                         logger.fine("Excluding schema " + schema +

198:                 } else {
199:                     if (fineEnabled) {
