 29: /**
 30:  * <tt>ModularAuthenticationStrategy</tt> implementation that requires <em>all</em> configured realms to
 31:  * <b>successfully</b> process the submitted <tt>AuthenticationToken</tt> during the log-in attempt.
 32:  *
 33:  * <p>If one or more realms do not support the submitted token, or one or more are unable to acquire
 34:  * <tt>AuthenticationInfo</tt> for the token, this implementation will immediately fail the log-in attempt for the
 35:  * associated subject (user).
 36:  *
 37:  * @author Les Hazlewood
 38:  * @since 0.2
 39:  */
 40: public class AllSuccessfulModularAuthenticationStrategy extends AbstractAuthenticationStrategy {
 41: 
 42:     /** Private class log instance. */
 43:     private static final Log log = LogFactory.getLog(AllSuccessfulModularAuthenticationStrategy.class);
 44: 
 45:     /**
 46:      * Because all realms in this strategy must complete successfully, this implementation ensures that the given
 47:      * <code>Realm</code> {@link Realm#supports(org.jsecurity.authc.AuthenticationToken) supports} the given
 48:      * <code>token</code> argument.  If it does not, this method throws an
 49:      * {@link org.jsecurity.authc.pam.UnsupportedTokenException UnsupportedTokenException} to end the authentication
 50:      * process immediately. If the realm does support the token, the <code>info</code> argument is returned immediately.
 51:      */
 52:     public AuthenticationInfo beforeAttempt(Realm realm, AuthenticationToken token, AuthenticationInfo info) throws AuthenticationException {
 53:         if (!realm.supports(token)) {
 54:             String msg = "Realm [" + realm + "] of type [" + realm.getClass().getName() + "] does not support " +
 55:                     " the submitted AuthenticationToken [" + token + "].  The [" + getClass().getName() +
 56:                     "] implementation requires all configured realm(s) to support and be able to process the submitted " +
 57:                     "AuthenticationToken.";
 58:             throw new UnsupportedTokenException(msg);
 59:         }
 60: 
 61:         return info;
 62:     }
 63: 
 64:     /**
 65:      * Merges the specified <code>info</code> into the <code>aggregate</code> argument and returns it (just as the
 66:      * parent implementation does), but additionally ensures the following:
 67:      * <ol>
 68:      * <li>if the <code>Throwable</code> argument is not <code>null</code>, re-throws it to immediately cancel the
 69:      * authentication process, since this strategy requires all realms to authenticate successfully.</li>
 70:      * <li>neither the <code>info</code> or <code>aggregate</code> argument is <code>null</code> to ensure that each
 71:      * realm did in fact authenticate successfully</li>
 72:      * </ol>
 73:      */
 74:     public AuthenticationInfo afterAttempt(Realm realm, AuthenticationToken token, AuthenticationInfo info, AuthenticationInfo aggregate, Throwable t)
 75:             throws AuthenticationException {
 76:         if (t != null) {
 77:             if (t instanceof AuthenticationException) {
 78:                 //propagate:
 79:                 throw ((AuthenticationException) t);
 80:             } else {
 81:                 String msg = "Unable to acquire account data from realm [" + realm + "].  The [" +
 82:                         getClass().getName() + " implementation requires all configured realm(s) to operate successfully " +
 83:                         "for a successful authentication.";
 84:                 throw new AuthenticationException(msg, t);
 85:             }
 86:         }
 87:         if (info == null) {
 88:             String msg = "Realm [" + realm + "] could not find any associated account data for the submitted " +
 89:                     "AuthenticationToken [" + token + "].  The [" + getClass().getName() + "] implementation requires " +
 90:                     "all configured realm(s) to acquire valid account data for a submitted token during the " +
 91:                     "log-in process.";
 92:             throw new UnknownAccountException(msg);
 93:         }
 94: 
 95:         // If non-null account is returned, then the realm was able to authenticate the
 96:         // user - so merge the account with any accumulated before:
 97:         if (log.isDebugEnabled()) {
 98:             log.debug("Account successfully authenticated using realm of type [" + realm.getClass().getName() + "]");
 99:         }
100: 
101:         merge(info, aggregate);
102: 
103:         return aggregate;
104:     }
105: }
