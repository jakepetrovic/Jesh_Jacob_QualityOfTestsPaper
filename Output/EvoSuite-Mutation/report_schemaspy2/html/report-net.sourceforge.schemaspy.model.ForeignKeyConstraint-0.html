 37:     private final String name;
 38:     private Table parentTable;
 39:     private final List<TableColumn> parentColumns = new ArrayList<TableColumn>();
 40:     private final Table childTable;
 41:     private final List<TableColumn> childColumns = new ArrayList<TableColumn>();
 42:     private final int deleteRule;
 43:     private final int updateRule;
 44:     private final static Logger logger = Logger.getLogger(ForeignKeyConstraint.class.getName());
 45:     private final static boolean finerEnabled = logger.isLoggable(Level.FINER);
 46: 
 47:     /**
 48:      * Construct a foreign key for the specified child table.
 49:      * Relationship details will be added later.
 50:      *
 51:      * @param child
 52:      * @param name
 53:      * @param deleteRule
 54:      */
 55:     ForeignKeyConstraint(Table child, String name, int updateRule, int deleteRule) {
 56:         this.name = name; // implied constraints will have a null name and override getName()
 57:         if (finerEnabled)
 58:             logger.finer("Adding foreign key constraint '" + getName() + "' to " + child.getFullName());
 59:         childTable = child;
 60:         this.deleteRule = deleteRule;
 61:         this.updateRule = updateRule;
 62:     }

 64:     /**
 65:      * This constructor is intended for use <b>after</b> all of the tables have been
 66:      * found in the system.  One impact of using this constructor is that it will
 67:      * "glue" the two tables together through their columns.
 68:      *
 69:      * @param parentColumn
 70:      * @param childColumn
 71:      */
 72:     public ForeignKeyConstraint(TableColumn parentColumn, TableColumn childColumn,
 73:                                 int updateRule, int deleteRule) {
 74:         this(childColumn.getTable(), null, updateRule, deleteRule);
 75: 
 76:         addChildColumn(childColumn);
 77:         addParentColumn(parentColumn);
 78: 
 79:         childColumn.addParent(parentColumn, this);
 80:         parentColumn.addChild(childColumn, this);
 81:     }

 83:     /**
 84:      * Same as {@link #ForeignKeyConstraint(TableColumn, TableColumn, int, int)},
 85:      * but defaults updateRule and deleteRule to

 87:      *
 88:      * @param parentColumn
 89:      * @param childColumn
 90:      */
 91:     public ForeignKeyConstraint(TableColumn parentColumn, TableColumn childColumn) {
 92:         this(parentColumn, childColumn, importedKeyNoAction, importedKeyNoAction);

289:     /**
290:      * Custom comparison method to deal with foreign key names that aren't
291:      * unique across all schemas being evaluated
292:      *
293:      * @param other ForeignKeyConstraint
294:      *
295:      * @return
296:      */
297:     public int compareTo(ForeignKeyConstraint other) {
298:         if (other == this)
299:             return 0;
300: 
301:         int rc = getName().compareToIgnoreCase(other.getName());
302:         if (rc == 0) {
303:             // should only get here if we're dealing with cross-schema references (rare)
304:             String ours = getChildColumns().get(0).getTable().getContainer();
305:             String theirs = other.getChildColumns().get(0).getTable().getContainer();
306:             if (ours != null && theirs != null)
307:                 rc = ours.compareToIgnoreCase(theirs);
308:             else if (ours == null)
309:                 rc = -1;
310:             else
311:                 rc = 1;
312:         }
313: 
314:         return rc;
315:     }
