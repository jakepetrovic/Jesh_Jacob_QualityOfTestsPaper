 64:      * @throws SQLException

 67:         this.table = table;
 68: 
 69:         // names and types are typically reused *many* times in a database,
 70:         // so keep a single instance of each distinct one
 71:         // (thanks to Mike Barnes for the suggestion)
 72:         String tmp = rs.getString("COLUMN_NAME");
 73:         name = tmp == null ? null : tmp.intern();
 74:         tmp = rs.getString("TYPE_NAME");
 75:         type = tmp == null ? "unknown" : tmp.intern();
 76: 
 77:         decimalDigits = rs.getInt("DECIMAL_DIGITS");
 78:         Number bufLength = (Number)rs.getObject("BUFFER_LENGTH");
 79:         if (bufLength != null && bufLength.shortValue() > 0)
 80:             length = bufLength.shortValue();
 81:         else
 82:             length = rs.getInt("COLUMN_SIZE");
 83: 
 84:         StringBuilder buf = new StringBuilder();
 85:         buf.append(length);
 86:         if (decimalDigits > 0) {
 87:             buf.append(',');
 88:             buf.append(decimalDigits);
 89:         }
 90:         detailedSize = buf.toString();
 91: 
 92:         isNullable = rs.getInt("NULLABLE") == DatabaseMetaData.columnNullable;
 93:         defaultValue = rs.getString("COLUMN_DEF");
 94:         setComments(rs.getString("REMARKS"));
 95:         id = new Integer(rs.getInt("ORDINAL_POSITION") - 1);
 96: 
 97:         Pattern excludeIndirectColumns = Config.getInstance().getIndirectColumnExclusions();
 98:         Pattern excludeColumns = Config.getInstance().getColumnExclusions();
