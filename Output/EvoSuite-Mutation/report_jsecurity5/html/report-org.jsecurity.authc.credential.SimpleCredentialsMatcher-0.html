 29: /**
 30:  * Simple CredentialsMatcher implementation.  Supports direct (plain) comparison for credentials of type
 31:  * byte[], char[], and Strings, and if the arguments do not match these types, then reverts back to simple
 32:  * <code>Object.equals</code> comparison.
 33:  *
 34:  * <p>Hashing comparisons (the most common technique used in secure applications) are not supported by this class, but
 35:  * instead by {@link HashedCredentialsMatcher HashedCredentialsMatcher} implementations.
 36:  *
 37:  * @author Les Hazlewood
 38:  * @see HashedCredentialsMatcher
 39:  * @see Md5CredentialsMatcher
 40:  * @see Sha1CredentialsMatcher
 41:  * @since 0.9
 42:  */
 43: public class SimpleCredentialsMatcher extends CodecSupport implements CredentialsMatcher {
 44: 
 45:     private static final Log log = LogFactory.getLog(SimpleCredentialsMatcher.class);
 46: 
 47:     /**
 48:      * Returns the <tt>token</tt>'s credentials.
 49:      *
 50:      * <p>This default implementation merely returns
 51:      * {@link AuthenticationToken#getCredentials() authenticationToken.getCredentials()} and exists as a template hook
 52:      * if subclasses wish to obtain the credentials in a different way or convert them to a different format before
 53:      * returning.
 54:      *
 55:      * @param token the <tt>AuthenticationToken</tt> submitted during the authentication attempt.
 56:      * @return the <tt>token</tt>'s associated credentials.
 57:      */
 58:     protected Object getCredentials(AuthenticationToken token) {
 59:         return token.getCredentials();
 60:     }
 61: 
 62:     /**
 63:      * Returns the <tt>account</tt>'s credentials.
 64:      *
 65:      * <p>This default implementation merely returns
 66:      * {@link AuthenticationInfo#getCredentials() account.getCredentials()} and exists as a template hook if subclasses
 67:      * wish to obtain the credentials in a different way or convert them to a different format before
 68:      * returning.
 69:      *
 70:      * @param info the <tt>AuthenticationInfo</tt> stored in the data store to be compared against the submitted authentication
 71:      *                token's credentials.
 72:      * @return the <tt>account</tt>'s associated credentials.
 73:      */
 74:     protected Object getCredentials(AuthenticationInfo info) {
 75:         return info.getCredentials();
 76:     }
 77: 
 78:     /**
 79:      * Returns <tt>true</tt> if the <tt>tokenCredentials</tt> argument is logically equal to the
 80:      * <tt>accountCredentials</tt> argument.
 81:      *
 82:      * <p>If both arguments are either a byte array (byte[]), char array (char[]) or String, they will be both be
 83:      * converted to raw byte arrays via the {@link #toBytes toBytes} method first, and then resulting byte arrays
 84:      * are compared via {@link Arrays#equals(byte[], byte[]) Arrays.equals(byte[],byte[])}.</p>
 85:      *
 86:      * <p>If either argument cannot be converted to a byte array as described, a simple Object <code>equals</code>
 87:      * comparison is made.</p>
 88:      *
 89:      * <p>Subclasses should override this method for more explicit equality checks.
 90:      *
 91:      * @param tokenCredentials   the <tt>AuthenticationToken</tt>'s associated credentials.
 92:      * @param accountCredentials the <tt>AuthenticationInfo</tt>'s stored credentials.
 93:      * @return <tt>true</tt> if the <tt>tokenCredentials</tt> are equal to the <tt>accountCredentials</tt>.
 94:      */
 95:     protected boolean equals(Object tokenCredentials, Object accountCredentials) {
 96:         if (log.isDebugEnabled()) {
 97:             log.debug("Performing credentials equality check for tokenCredentials of type [" +
 98:                     tokenCredentials.getClass().getName() + " and accountCredentials of type [" +
 99:                     accountCredentials.getClass().getName() + "]");
100:         }
101:         if ((tokenCredentials instanceof byte[] || tokenCredentials instanceof char[] || tokenCredentials instanceof String) &&
102:                 (accountCredentials instanceof byte[] || accountCredentials instanceof char[] || accountCredentials instanceof String)) {
103:             if (log.isDebugEnabled()) {
104:                 log.debug("Both credentials arguments can be easily converted to byte arrays.  Performing " +
105:                         "array equals comparison");
106:             }
107:             byte[] tokenBytes = toBytes(tokenCredentials);
108:             byte[] accountBytes = toBytes(accountCredentials);
109:             return Arrays.equals(tokenBytes, accountBytes);
110:         } else {
111:             return accountCredentials.equals(tokenCredentials);
112:         }
113:     }
114: 
115:     /**
116:      * This implementation acquires the <tt>token</tt>'s credentials
117:      * (via {@link #getCredentials(AuthenticationToken) getCredentials(token)})
118:      * and then the <tt>account</tt>'s credentials
119:      * (via {@link #getCredentials(org.jsecurity.authc.AuthenticationInfo) getCredentials(account)}) and then passes both of
120:      * them to the {@link #equals(Object,Object) equals(tokenCredentials, accountCredentials)} method for equality
121:      * comparison.
122:      *
123:      * @param token   the <tt>AuthenticationToken</tt> submitted during the authentication attempt.
124:      * @param info the <tt>AuthenticationInfo</tt> stored in the system matching the token principal.
125:      * @return <tt>true</tt> if the provided token credentials are equal to the stored account credentials,
126:      *         <tt>false</tt> otherwise
127:      */
128:     public boolean doCredentialsMatch(AuthenticationToken token, AuthenticationInfo info) {
129:         Object tokenCredentials = getCredentials(token);
130:         Object accountCredentials = getCredentials(info);
131:         return equals(tokenCredentials, accountCredentials);
132:     }
