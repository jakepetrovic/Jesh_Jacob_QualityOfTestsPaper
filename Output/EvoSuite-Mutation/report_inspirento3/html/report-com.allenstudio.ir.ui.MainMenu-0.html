 65:     private JMenu createMenu(String menu) {
 66:         JMenu jMenu = null;
 67:                 
 68:         try {
 69:             String tempLabel = UIResources.getString(menu + LABEL_SUFFIX);
 70:             if (tempLabel.length() > 2 
 71:                     && tempLabel.charAt(tempLabel.length() - 2) == '*') {
 72:                 //Add Mnemonic
 73:                 jMenu = new JMenu(tempLabel.substring(0, tempLabel.length() - 2));
 74:                 jMenu.setMnemonic(tempLabel.charAt(tempLabel.length() - 1));
 75:             }
 76:             else {
 77:                 jMenu = new JMenu(tempLabel);
 78:             }
 79:             
 80:             String[] items = InspirentoUtilities.tokenize(UIResources.getString(menu));
 81:             for (String item : items) {
 82:                 if (!("-".equals(item))) {
 83:                     if (item.length() > 1 &&
 84:                             item.charAt(item.length() - 1) == '>') {
 85: //                        Add submenus recursively
 86:                         jMenu.add(createMenu(item.substring(0, item.length() - 1)));
 87:                     }
 88:                     else if (item.charAt(0) == '#') {//CheckBoxMenuItem
 89: //                      Actions should be binded with menu items in this block
 90:                         String itemLabel = UIResources.getString(item.substring(1)
 91:                                 + LABEL_SUFFIX);
 92:                         JCheckBoxMenuItem jMenuItem = null;
 93:                         if (itemLabel.length() > 2 
 94:                                 && itemLabel.charAt(itemLabel.length() - 2) == '*') {
 95:     //                      Add Mnemonic
 96:                             jMenuItem = new JCheckBoxMenuItem(itemLabel.substring(0, itemLabel.length() - 2));
 97:                             jMenuItem.setMnemonic(itemLabel.charAt(itemLabel.length() - 1));
 98:                         }
 99:                         else {
100:                             jMenuItem = new JCheckBoxMenuItem(itemLabel);
101:                         }
102:     //                  Add icon and shortcut
103:                         addShortcutAndIcon(jMenuItem, item);
104:                         
105:                         jMenu.add(jMenuItem);
106:                     }
107:                     else {
108: //                        Actions should be binded with menu items in this block
109:                         String itemLabel = UIResources.getString(item + LABEL_SUFFIX);
110:                         JMenuItem jMenuItem = null;
