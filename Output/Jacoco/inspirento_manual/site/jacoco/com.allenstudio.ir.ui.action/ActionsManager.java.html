package com.allenstudio.ir.ui.action;

/**
 * Manages all actions registered with
 * certain buttons or menu items.
 * This approach separates the action from
 * user interface.<br>For future updating
 * and modifying purpose, I intend to
 * use an XML file to store various actions.
 * If this is done, synchronization must be
 * noticed to make sure that the contents in
 * this file is compatible with that in the ui
 * resource file. For instance, if a menu item
 * named "Find" is defined in the resource file,
 * but there no concern in the XML action preset
 * file, the item would not interact with the user.
 * I think forName() method should be used.
 * 
 * @author Allen Chue
 */
public class ActionsManager {
    private ActionsManager instance = null;
    
    private ActionsManager() {
        
    }
    
    /**
     * Singleton method. Because this
     * action manager should be unique. 
     * @return an <code>ActionsManager</code> instance
     */
    public ActionsManager getInstance() {
        if (instance != null) {
            return instance;
        }
        else {
            instance = new ActionsManager();
            return instance;
        }
    }
}
