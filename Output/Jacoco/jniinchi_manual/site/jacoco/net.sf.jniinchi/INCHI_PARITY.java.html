package net.sf.jniinchi;

/**
 * Enumeration of InChI 0D parities.
 * Corresponds to <tt>inchi_StereoParity0D</tt> in <tt>inchi_api.h</tt>.
 * @author Sam Adams
 */
public enum INCHI_PARITY {

    /**
     * None.
     */
    NONE(0),

    /**
     * Odd.
     */
    ODD(1),

    /**
     * Even.
     */
    EVEN(2),

    /**
     * Unknown.
     */
    UNKNOWN(3),

    /**
     * Undefined.
     */
    UNDEFINED(4);


    /**
     * Internal InChI index (from inchi_api.h).
     */
    private final int indx;

    /**
     * Constructor.
     */
    private INCHI_PARITY(final int indx) {
        this.indx = indx;
    }

    public int getIndx() {
        return indx;
    }


    public static INCHI_PARITY getValue(int parity) {
        switch (parity) {
            case 0:
                return NONE;
            case 1:
                return ODD;
            case 2:
                return EVEN;
            case 3:
                return UNKNOWN;
            case 4:
                return UNDEFINED;
            default:
                return null;
        }
    }
}
